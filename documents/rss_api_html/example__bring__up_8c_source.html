21 /** \example example_bring_up.c
22  * @brief This is an example on how the assembly test can be used to ease bring-up
23  * @n
24  * The example executes as follows:
25  * - Register the RSS HAL
26  * - Create assembly test
27  * - Enable one assembly test at a time
28  * - Power on the sensor
29  * - Enable the sensor
30  * - Exceute assembly test
31  * - Disable the sensor
32  * - Power off the sensor
33  * - Destroy assembly test
34  */

47 int acconeer_main(int argc, char *argv[])
48 {
49  (void)argc;
50  (void)argv;
51 
52  bool test_ok = true;
53 
54  printf("Acconeer software version %s\n", acc_version_get());
55 
56  const acc_hal_a121_t *hal = acc_hal_rss_integration_get_implementation();
57 
58  if (!acc_rss_hal_register(hal))
59  {
60  return EXIT_FAILURE;
61  }
62 
63  void *buffer = acc_integration_mem_alloc(ACC_RSS_ASSEMBLY_TEST_MIN_BUFFER_SIZE);
64  if (buffer == NULL)
65  {
66  printf("Assembly test: Memory allocation failed\n");
67  return EXIT_FAILURE;
68  }
69 
70  // Create
71  acc_rss_assembly_test_t *assembly_test = acc_rss_assembly_test_create(SENSOR_ID, buffer, ACC_RSS_ASSEMBLY_TEST_MIN_BUFFER_SIZE);
72  if (assembly_test == NULL)
73  {
74  printf("Bring-up: Could not create assembly test\n");
75  acc_integration_mem_free(buffer);
76  return EXIT_FAILURE;
77  }
78 
79  // Enable and run: Basic read test
80  acc_rss_assembly_test_disable_all_tests(assembly_test);
81  acc_rss_assembly_test_enable(assembly_test, ACC_RSS_ASSEMBLY_TEST_ID_BASIC_READ);
82  if (!run_test(assembly_test, SENSOR_ID))
83  {
84  printf("Bring-up: Basic read test failed\n");
85  test_ok = false;
86  }
87 
88  // Enable and run: Communication test
89  acc_rss_assembly_test_disable_all_tests(assembly_test);
90  acc_rss_assembly_test_enable(assembly_test, ACC_RSS_ASSEMBLY_TEST_ID_COMMUNICATION);
91  if (!run_test(assembly_test, SENSOR_ID))
92  {
93  printf("Bring-up: Communication test failed\n");
94  test_ok = false;
95  }
96 
97  // Enable and run: Enable test
98  acc_rss_assembly_test_disable_all_tests(assembly_test);
99  acc_rss_assembly_test_enable(assembly_test, ACC_RSS_ASSEMBLY_TEST_ID_ENABLE_PIN);
100  if (!run_test(assembly_test, SENSOR_ID))
101  {
102  printf("Bring-up: Enable test failed\n");
103  test_ok = false;
104  }
105 
106  // Enable and run: Interrupt test
107  acc_rss_assembly_test_disable_all_tests(assembly_test);
108  acc_rss_assembly_test_enable(assembly_test, ACC_RSS_ASSEMBLY_TEST_ID_INTERRUPT);
109  if (!run_test(assembly_test, SENSOR_ID))
110  {
111  printf("Bring-up: Interrupt test failed\n");
112  test_ok = false;
113  }
114 
115  // Enable and run: Clock and Supply test
116  acc_rss_assembly_test_disable_all_tests(assembly_test);
117  acc_rss_assembly_test_enable(assembly_test, ACC_RSS_ASSEMBLY_TEST_ID_CLOCK_AND_SUPPLY);
118  if (!run_test(assembly_test, SENSOR_ID))
119  {
120  printf("Bring-up: Clock and Supply test failed\n");
121  test_ok = false;
122  }
123 
124  // Enable and run: Sensor calibration test
125  acc_rss_assembly_test_disable_all_tests(assembly_test);
126  acc_rss_assembly_test_enable(assembly_test, ACC_RSS_ASSEMBLY_TEST_ID_SENSOR_CALIBRATION);
127  if (!run_test(assembly_test, SENSOR_ID))
128  {
129  printf("Bring-up: Calibration test failed\n");
130  test_ok = false;
131  }
132 
133  // Destroy
134  acc_rss_assembly_test_destroy(assembly_test);
135 
136  if (buffer != NULL)
137  {
138  acc_integration_mem_free(buffer);
139  }
140 
141  if (test_ok)
142  {
143  printf("Bring-up: All tests passed\n");
144 
145  printf("Application finished OK\n");
146  return EXIT_SUCCESS;
147  }
148 
149  return EXIT_SUCCESS;
150 }

153 static bool run_test(acc_rss_assembly_test_t *assembly_test, acc_sensor_id_t sensor_id)
154 {
155  bool all_passed = true;
156 
157  acc_hal_integration_sensor_supply_on(sensor_id);
158  acc_hal_integration_sensor_enable(sensor_id);
159 
160  acc_rss_test_state_t assembly_test_state = ACC_RSS_TEST_STATE_ONGOING;
161  acc_rss_test_integration_status_t integration_status = ACC_RSS_TEST_INTEGRATION_STATUS_OK;
162 
163  do
164  {
165  assembly_test_state = acc_rss_assembly_test_execute(assembly_test, integration_status);
166 
167  switch (assembly_test_state)
168  {
169  case ACC_RSS_TEST_STATE_TOGGLE_ENABLE_PIN:
170  acc_hal_integration_sensor_disable(sensor_id);
171  acc_hal_integration_sensor_enable(sensor_id);
172  integration_status = ACC_RSS_TEST_INTEGRATION_STATUS_OK;
173  break;
174  case ACC_RSS_TEST_STATE_WAIT_FOR_INTERRUPT:
175  if (!acc_hal_integration_wait_for_sensor_interrupt(sensor_id, SENSOR_TIMEOUT_MS))
176  {
177  /* Wait for interrupt failed */
178  integration_status = ACC_RSS_TEST_INTEGRATION_STATUS_TIMEOUT;
179  }
180  else
181  {
182  integration_status = ACC_RSS_TEST_INTEGRATION_STATUS_OK;
183  }
184 
185  break;
186  default:
187  integration_status = ACC_RSS_TEST_INTEGRATION_STATUS_OK;
188  break;
189  }
190  } while (assembly_test_state != ACC_RSS_TEST_STATE_COMPLETE);
191 
192  acc_hal_integration_sensor_disable(sensor_id);
193  acc_hal_integration_sensor_supply_off(sensor_id);
194 
195  uint16_t nbr_of_test_results = 0U;
196 
197  const acc_rss_assembly_test_result_t *test_results = acc_rss_assembly_test_get_results(assembly_test, &nbr_of_test_results);
198 
199  for (uint16_t idx = 0; idx < nbr_of_test_results; idx++)
200  {
201  printf("Bring-up: '%s' [%s]\n", test_results[idx].test_name, test_results[idx].test_result ? "PASS" : "FAIL");
202  if (!test_results[idx].test_result)
203  {
204  all_passed = false;
205  }
206  }
207 
208  return all_passed;
209 }
