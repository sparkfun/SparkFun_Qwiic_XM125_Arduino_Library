21 /** \example example_diagnostic_test.c
22  * @brief This is an example on how to use RSS to execute the diagnostic test
23  * @n
24  * The example executes as follows:
25  * - Register the RSS HAL
26  * - Power on the sensor
27  * - Create diagnostic test
28  * - Enable the sensor
29  * - Execute diagnostic test
30  * - Destroy diagnostic test
31  * - Disable the sensor
32  * - Power off the sensor
33  */
34 
35 #define SENSOR_ID (1U)
36 
37 #define SENSOR_TIMEOUT_MS (1000U)

42 int acconeer_main(int argc, char *argv[])
43 {
44  (void)argc;
45  (void)argv;
46 
47  bool test_ok = true;
48 
49  printf("Acconeer software version %s\n", acc_version_get());
50 
51  const acc_hal_a121_t *hal = acc_hal_rss_integration_get_implementation();
52 
53  if (!acc_rss_hal_register(hal))
54  {
55  return EXIT_FAILURE;
56  }
57 
58  void *buffer = acc_integration_mem_alloc(ACC_RSS_ASSEMBLY_TEST_MIN_BUFFER_SIZE);
59  if (buffer == NULL)
60  {
61  printf("Diagnostic test: Memory allocation failed\n");
62  return EXIT_FAILURE;
63  }
64 
65  acc_rss_test_state_t test_state = ACC_RSS_TEST_STATE_ONGOING;
66  acc_rss_test_integration_status_t integration_status = ACC_RSS_TEST_INTEGRATION_STATUS_OK;
67 
68  acc_rss_assembly_test_t *assembly_test = acc_rss_assembly_test_create(SENSOR_ID, buffer, ACC_RSS_ASSEMBLY_TEST_MIN_BUFFER_SIZE);
69 
70  if (assembly_test != NULL)
71  {
72  /* Enable all tests */
73  acc_rss_assembly_test_enable_all_tests(assembly_test);
74 
75  /* Enable the diagnostic logs from the assembly test */
76  acc_rss_assembly_test_enable_diagnostic_logs();
77 
78  acc_hal_integration_sensor_supply_on(SENSOR_ID);
79  acc_hal_integration_sensor_enable(SENSOR_ID);
80 
81  do
82  {
83  test_state = acc_rss_assembly_test_execute(assembly_test, integration_status);
84 
85  switch (test_state)
86  {
87  case ACC_RSS_TEST_STATE_TOGGLE_ENABLE_PIN:
88  acc_hal_integration_sensor_disable(SENSOR_ID);
89  acc_hal_integration_sensor_enable(SENSOR_ID);
90  integration_status = ACC_RSS_TEST_INTEGRATION_STATUS_OK;
91  break;
92  case ACC_RSS_TEST_STATE_WAIT_FOR_INTERRUPT:
93  if (!acc_hal_integration_wait_for_sensor_interrupt(SENSOR_ID, SENSOR_TIMEOUT_MS))
94  {
95  /* Wait for interrupt failed */
96  integration_status = ACC_RSS_TEST_INTEGRATION_STATUS_TIMEOUT;
97  }
98 
99  break;
100  default:
101  integration_status = ACC_RSS_TEST_INTEGRATION_STATUS_OK;
102  break;
103  }
104  } while (test_state != ACC_RSS_TEST_STATE_COMPLETE);
105 
106  acc_rss_assembly_test_destroy(assembly_test);
107 
108  acc_hal_integration_sensor_disable(SENSOR_ID);
109  acc_hal_integration_sensor_supply_off(SENSOR_ID);
110  }
111  else
112  {
113  printf("Diagnostic test: Could not create assembly test\n");
114  test_ok = false;
115  }
116 
117  if (buffer != NULL)
118  {
119  acc_integration_mem_free(buffer);
120  }
121 
122  if (test_ok)
123  {
124  printf("Diagnostic test: All tests passed\n");
125 
126  printf("Application finished OK\n");
127  return EXIT_SUCCESS;
128  }
129 
130  return EXIT_FAILURE;
131 }
