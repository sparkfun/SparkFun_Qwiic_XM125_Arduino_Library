265 static inline void prepare_power_down(void)
266 {
267  HAL_RCC_GetOscConfig(&low_power_saved_rcc_oscinitstruct);
268  HAL_RCC_GetClockConfig(&low_power_saved_rcc_clkinitstruct, &low_power_saved_flatency);
269 
270  HAL_I2C_DeInit(&MODULE_I2C_HANDLE);
271  HAL_SPI_DeInit(&A121_SPI_HANDLE);
272  HAL_UART_DeInit(&MODULE_UART1_HANDLE);
273 
274  /* Save GPIO bank state */
275  for (size_t idx = 0; idx < GPIO_BANK_COUNT; idx++)
276  {
277  GPIO_TypeDef *gpio_bank = gpio_banks[idx];
278  gpio_config_t *gpio_config = &low_power_saved_gpio_bank[idx];
279  gpio_config->MODER = READ_REG(gpio_bank->MODER);
280  gpio_config->OTYPER = READ_REG(gpio_bank->OTYPER);
281  gpio_config->OSPEEDR = READ_REG(gpio_bank->OSPEEDR);
282  gpio_config->PUPDR = READ_REG(gpio_bank->PUPDR);
283  gpio_config->AFR[0] = READ_REG(gpio_bank->AFR[0]);
284  gpio_config->AFR[1] = READ_REG(gpio_bank->AFR[1]);
285  }
286 
287  /* Turn GPIO off */
288  GPIO_InitTypeDef GPIO_InitStructOff;
289 
290  /* Set all unused GPIO pins in the lowest power consuming state according to AN4899*/
291  GPIO_InitStructOff.Mode = GPIO_MODE_ANALOG;
292  GPIO_InitStructOff.Pull = GPIO_NOPULL;
293 
294  /* GPIO A */
295  GPIO_InitStructOff.Pin = GPIO_PIN_All;
296  GPIO_InitStructOff.Pin &= (uint16_t) ~(WAKE_UP_Pin | SPI_SCK_Pin | SPI_MISO_Pin | SPI_MOSI_Pin);
297  HAL_GPIO_Init(GPIOA, &GPIO_InitStructOff);
298 
299  /* GPIO B */
300  GPIO_InitStructOff.Pin = GPIO_PIN_All;
301  GPIO_InitStructOff.Pin &= (uint16_t) ~(MCU_INT_Pin | ENABLE_Pin | SPI_SS_Pin | MISC_GPIO0_Pin);
302  HAL_GPIO_Init(GPIOB, &GPIO_InitStructOff);
303 
304  /* GPIO H */
305  GPIO_InitStructOff.Pin = GPIO_PIN_All;
306  HAL_GPIO_Init(GPIOH, &GPIO_InitStructOff);
307 
308  /* Disable Clocks */
309  __HAL_RCC_GPIOA_CLK_DISABLE();
310  __HAL_RCC_GPIOB_CLK_DISABLE();
311  __HAL_RCC_GPIOH_CLK_DISABLE();
312 
313  HAL_SuspendTick();
314 }

331  GPIO_TypeDef *gpio_bank = gpio_banks[idx];

344  HAL_UART_Init(&MODULE_UART2_HANDLE);
345 
346  /* Start I2C */

358 static uint32_t get_rtc_tick(void)
359 {
360  RTC_DateTypeDef rtc_date = { 0 };
361  RTC_TimeTypeDef rtc_time = { 0 };
362 
363  /* Wait until any pending shift operation is completed */
364  while ((MODULE_RTC_HANDLE.Instance->ISR & RTC_ISR_SHPF) != RESET)
365  {
366  ;
367  }
368 
369  if (HAL_RTC_GetTime(&MODULE_RTC_HANDLE, &rtc_time, RTC_FORMAT_BIN) != HAL_OK)
370  {
371  Error_Handler();
372  }
373 
374  if (HAL_RTC_GetDate(&MODULE_RTC_HANDLE, &rtc_date, RTC_FORMAT_BIN) != HAL_OK)
375  {
376  Error_Handler();
377  }
378 
379  uint32_t rtc_ticks_ms = 0;
380 
381  if (rtc_time.Hours)
382  {
383  rtc_ticks_ms += rtc_time.Hours * 60 * 60 * 1000;
384  }
385 
386  if (rtc_time.Minutes)
387  {
388  rtc_ticks_ms += rtc_time.Minutes * 60 * 1000;
389  }
390 
391  if (rtc_time.Seconds)
392  {
393  rtc_ticks_ms += rtc_time.Seconds * 1000;
394  }
395 
396  rtc_ticks_ms += ((rtc_time.SecondFraction - rtc_time.SubSeconds) * 1000) / (rtc_time.SecondFraction + 1);
397 
398  return rtc_ticks_ms;
399 }

402 static void rtc_tick_to_time(uint32_t tick, RTC_TimeTypeDef *time)
403 {
404  uint32_t rtc_ticks_ms = tick;
405 
406  time->SecondFraction = MODULE_RTC_HANDLE.Init.SynchPrediv;
407 
408  time->Hours = (rtc_ticks_ms / (60 * 60 * 1000)) % 24;
409 
410  rtc_ticks_ms = rtc_ticks_ms % (60 * 60 * 1000);
411 
412  time->Minutes = (rtc_ticks_ms / (60 * 1000)) % 60;
413 
414  rtc_ticks_ms = rtc_ticks_ms % (60 * 1000);
415 
416  time->Seconds = (rtc_ticks_ms / 1000) % 60;
417 
418  rtc_ticks_ms = rtc_ticks_ms % 1000;
419 
420  time->SubSeconds = time->SecondFraction - (rtc_ticks_ms * (time->SecondFraction + 1)) / 1000;
421 }

424 static void rtc_set_next_wakeup_time(void)
425 {
426  RTC_AlarmTypeDef alarm = { {0}, 0, 0, 0, 0, 0, 0};
427 
428  if (rtc_period_time_ms > 0)
429  {
430  rtc_tick_to_time(get_rtc_tick() + rtc_period_time_ms, &alarm.AlarmTime);
431 
432  alarm.Alarm = RTC_ALARM_B;
433  alarm.AlarmMask = RTC_ALARMMASK_DATEWEEKDAY;
434  alarm.AlarmSubSecondMask = RTC_ALARMSUBSECONDMASK_SS14;
435 
436  if (HAL_RTC_DeactivateAlarm(&MODULE_RTC_HANDLE, RTC_ALARM_B) != HAL_OK)
437  {
438  Error_Handler();
439  }
440 
441  if (HAL_RTC_SetAlarm_IT(&MODULE_RTC_HANDLE, &alarm, RTC_FORMAT_BIN) != HAL_OK)
442  {
443  Error_Handler();
444  }
445  }
446  else
447  {
448  if (HAL_RTC_DeactivateAlarm(&MODULE_RTC_HANDLE, RTC_ALARM_B) != HAL_OK)
449  {
450  Error_Handler();
451  }
452  }
