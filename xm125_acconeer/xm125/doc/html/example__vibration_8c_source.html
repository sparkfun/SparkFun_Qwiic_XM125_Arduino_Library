32 /** \example example_vibration.c
33  * @brief This is an example on how the Service API can be used to measure
34  * vibrations
35  * @n
36  * The example executes as follows:
37  * - Create a configuration
38  * - Create a processing instance using the previously created configuration
39  * - Create a sensor instance

41  * - Perform a sensor measurement and read out the data
42  * - Process the measurement and print vibration result
43  * - Check 'calibration_needed' indication
44  * - Destroy the sensor instance
45  * - Destroy the processing instance
46  * - Destroy the configuration

224 int acconeer_main(int argc, char *argv[])
225 {
226  (void)argc;
227  (void)argv;
228 
229  acc_vibration_config_t config = {0};
230  acc_vibration_app_t app = {0};
231  acc_vibration_result_t result = {0};
232 
233  printf("Acconeer software version %s\n", acc_version_get());
234 
235  const acc_hal_a121_t *hal = acc_hal_rss_integration_get_implementation();
236 
237  if (!acc_rss_hal_register(hal))
238  {
239  return EXIT_FAILURE;
240  }
241 
242  config.sensor_config = acc_config_create();
243  if (config.sensor_config == NULL)
244  {
245  printf("acc_config_create() failed\n");
246  cleanup(&app, &config);
247  return EXIT_FAILURE;
248  }
249 
250  set_config(&config);
251 
252  // Print the configuration
253  acc_config_log(config.sensor_config);
254 
255  if (!validate_config(&config))
256  {
257  printf("Invalid sensor config\n");
258  cleanup(&app, &config);
259  return EXIT_FAILURE;
260  }
261 
262  if (!init_sensor_resources(&app, &config))
263  {
264  printf("Failed to initialize sensor resources\n");
265  cleanup(&app, &config);
266  return EXIT_FAILURE;
267  }
268 
269  if (!init_app_resources(&app, &config))
270  {
271  printf("Failed to initialize application\n");
272  cleanup(&app, &config);
273  return EXIT_FAILURE;
274  }
275 
276  if (!do_sensor_calibration_and_prepare(&app, &config))
277  {
278  printf("do_sensor_calibration_and_prepare() failed\n");
279  acc_sensor_status(app.sensor);
280  cleanup(&app, &config);
281  return EXIT_FAILURE;
282  }
283 
284  while (true)
285  {
286  if (!do_sensor_measurement(&app))
287  {
288  printf("Failed to do sensor measurement\n");
289  cleanup(&app, &config);
290  return EXIT_FAILURE;
291  }
292 
293  if (app.proc_result.calibration_needed)
294  {
295  printf("The current calibration is not valid for the current temperature.\n");
296  printf("The sensor needs to be re-calibrated.\n");
297 
298  if (!do_sensor_calibration_and_prepare(&app, &config))
299  {
300  printf("do_sensor_calibration_and_prepare() failed\n");
301  acc_sensor_status(app.sensor);
302  cleanup(&app, &config);
303  return EXIT_FAILURE;
304  }
305 
306  printf("The sensor was successfully re-calibrated.\n");
307  }
308  else
309  {
310  process(&app, &config, &result);
311  print_result(&app, &result);
312  }
313  }
314 
315  cleanup(&app, &config);
316 
317  printf("Application finished OK\n");
318 
319  return EXIT_SUCCESS;
320 }

357 static bool validate_config(acc_vibration_config_t *config)
358 {
359  bool success = true;
360 
361  if (!IS_POWER_OF_TWO(config->time_series_length))
362  {
363  printf("**************************** [Warning] *********************************\n");
364  printf("** time_series_length should be power of 2 for efficient usage of FFT **\n");
365  printf("************************************************************************\n");
366  }
367 
368  if (config->time_series_length < 2U)
369  {
370  printf("time_series_length too small\n");
371  success = false;
372  }
373 
374  if (acc_config_sweep_rate_get(config->sensor_config) == 0.0f)
375  {
376  printf("sweep_rate must be set\n");
377  success = false;
378  }
379 
380  if (acc_config_num_points_get(config->sensor_config) != 1U)
381  {
382  printf("num_points must be set to 1\n");
383  success = false;
384  }
385 
386  if (acc_config_continuous_sweep_mode_get(config->sensor_config) && !acc_config_double_buffering_get(config->sensor_config))
387  {
388  printf("double buffering needs to be activated to achieve sufficient sweep rate when using continuous sweep mode\n");
389  success = false;
390  }
391 
392  return success;
393 }

432 static bool init_app_resources(acc_vibration_app_t *app, acc_vibration_config_t *config)
433 {
434  uint16_t sweeps_per_frame = acc_config_sweeps_per_frame_get(config->sensor_config);
435  uint16_t num_points = acc_config_num_points_get(config->sensor_config);
436  float sweep_rate = acc_config_sweep_rate_get(config->sensor_config);
437 
438  app->has_init = false;
439 
440  setup_rfft_bounds(app, config);
441 
442  app->frame_length = sweeps_per_frame * num_points;
443  app->rfft_output_length = (app->padded_time_series_length / 2U) + 1U;
444  app->rfft_read_offset = 1U;
445  app->data_length = app->rfft_output_length - app->rfft_read_offset;
446 
447  app->continuous_data_acquisition = acc_config_continuous_sweep_mode_get(config->sensor_config);
448 
449  /*
450  * speed_of_light = 299792458f
451  * radio_frequency = 60.5e9f
452  * perceived_wavelength = (speed_of_light / radio_frequency) / 2.0f
453  */
454  const float perceived_wavelength = 0.0024776f; // 0.002477623619834711
455 
456  app->radians_to_displacement = (perceived_wavelength * 1e6f) / (2.0f * (float)M_PI);
457 
458  app->psd_to_radians_conversion_factor =
459  2.0f / (app->continuous_data_acquisition ? (float)config->time_series_length : (float)sweeps_per_frame);
460 
461  app->displacement_conversion_factor = app->psd_to_radians_conversion_factor * app->radians_to_displacement;
462 
463  if (config->reported_displacement_mode == ACC_VIBRATION_REPORT_DISPLACEMENT_AS_PEAK2PEAK)
464  {
465  app->displacement_conversion_factor *= 2.0f;
466  }
467 
468  app->time_series = acc_integration_mem_calloc(config->time_series_length, sizeof(*app->time_series));
469  if (app->time_series == NULL)
470  {
471  printf("Failed to allocate memory for time_series array\n");
472  return false;
473  }
474 
475  app->frequencies = acc_integration_mem_calloc(app->rfft_output_length, sizeof(*app->frequencies));
476  if (app->frequencies == NULL)
477  {
478  printf("Failed to allocate memory for frequencies array\n");
479  return false;
480  }
481 
482  setup_sample_frequencies(app, sweep_rate);
483 
484  app->double_buffer_filter_buffer =
485  acc_integration_mem_calloc(sweeps_per_frame - 2U, sizeof(*app->double_buffer_filter_buffer));
486  if (app->double_buffer_filter_buffer == NULL)
487  {
488  printf("Failed to allocate memory for double_buffer_filter_buffer\n");
489  return false;
490  }
491 
492  app->zero_mean_time_series = acc_integration_mem_calloc(config->time_series_length, sizeof(*app->zero_mean_time_series));
493  if (app->zero_mean_time_series == NULL)
494  {
495  printf("Failed to allocate memory for zero_mean_time_series array\n");
496  return false;
497  }
498 
499  app->rfft_output = acc_integration_mem_calloc(app->rfft_output_length, sizeof(*app->rfft_output));
500  if (app->rfft_output == NULL)
501  {
502  printf("Failed to allocate memory for rfft_output array\n");
503  return false;
504  }
505 
506  app->lp_displacements = acc_integration_mem_calloc(app->data_length, sizeof(*app->lp_displacements));
507  if (app->lp_displacements == NULL)
508  {
509  printf("Failed to allocate memory for lp_displacements array\n");
510  return false;
511  }
512 
513  app->threshold = acc_integration_mem_calloc(app->data_length, sizeof(*app->threshold));
514  if (app->threshold == NULL)
515  {
516  printf("Failed to allocate memory for threshold\n");
517  return false;
518  }
519 
520  return true;
521 }

532 static bool do_sensor_calibration_and_prepare(acc_vibration_app_t *app, acc_vibration_config_t *config)
533 {
534  bool status = false;
535  bool cal_complete = false;
536  acc_cal_result_t cal_result = {0};
537 
538  const uint16_t calibration_retries = 1U;
539 
540  // Random disturbances may cause the calibration to fail. At failure, retry at least once.
541  for (uint16_t i = 0; !status && (i <= calibration_retries); i++)
542  {
543  // Reset sensor before calibration by disabling/enabling it
544  acc_hal_integration_sensor_disable(SENSOR_ID);
545  acc_hal_integration_sensor_enable(SENSOR_ID);
546 
547  do
548  {
549  status = acc_sensor_calibrate(app->sensor,
550  &cal_complete,
551  &cal_result,
552  app->rss_buffer,
553  app->rss_buffer_size);
554 
555  if (status && !cal_complete)
556  {
557  status = acc_hal_integration_wait_for_sensor_interrupt(SENSOR_ID, SENSOR_TIMEOUT_MS);
558  }
559  } while (status && !cal_complete);
560  }
561 
562  if (status)
563  {
564  // Reset sensor after calibration by disabling/enabling it
565  acc_hal_integration_sensor_disable(SENSOR_ID);
566  acc_hal_integration_sensor_enable(SENSOR_ID);
567 
568  status = acc_sensor_prepare(app->sensor,
569  config->sensor_config,
570  &cal_result,
571  app->rss_buffer,
572  app->rss_buffer_size);
573  }
574 
575  return status;
576 }

608 static void process(acc_vibration_app_t *app, acc_vibration_config_t *config, acc_vibration_result_t *result)
609 {
610  const uint16_t num_points = acc_config_num_points_get(config->sensor_config);
611  const uint16_t sweeps_per_frame = acc_config_sweeps_per_frame_get(config->sensor_config);
612 
613  result->max_sweep_amplitude = 0.0f;
614  result->max_displacement = FLT_MAX;
615  result->max_displacement_freq = FLT_MAX;
616  result->time_series_std = FLT_MAX;
617 
618  /*
619  * Check if an object is close enough in front of the sensor to
620  * reach the configured threshold.
621  */
622 
623  for (uint16_t i = 0; i < app->frame_length; i++)
624  {
625  acc_int16_complex_t point = app->proc_result.frame[i];
626  result->max_sweep_amplitude = fmaxf(result->max_sweep_amplitude, cabsf(point.real + (point.imag * I)));
627  }
628 
629  if (result->max_sweep_amplitude < config->amplitude_threshold)
630  {
631  // Return without a (complete) result.
632  app->has_init = false;
633  return;
634  }
635 
636  /*
637  * Process sensor data.
638  */
639 
640  // Handle frame based on whether or not continuous sweep mode is used.
641  if (app->continuous_data_acquisition)
642  {
643  acc_algorithm_double_buffering_frame_filter(app->proc_result.frame, sweeps_per_frame, num_points,
644  app->double_buffer_filter_buffer);
645  roll_and_update_time_series(app, config);
646  }
647  else
648  {
649  for (uint16_t i = 0; i < app->frame_length; i++)
650  {
651  acc_int16_complex_t point = app->proc_result.frame[i];
652  app->time_series[i] = cargf(point.real + (point.imag * I));
653  }

679  }

748 static void calculate_threshold(acc_vibration_app_t *app, acc_vibration_config_t *config)
749 {
750  for (uint16_t i = 0; i < app->data_length; i++)
751  {
752  app->threshold[i] = acc_algorithm_calculate_cfar(app->lp_displacements,
753  app->data_length,
754  CFAR_WINDOW_LENGTH,
755  CFAR_HALF_GUARD_LENGTH,
756  config->threshold_sensitivity,
757  i);
758  }
759 
760  /*
761  * Extend the CFAR threshold using extrapolation.
762  *
763  * The head is extended using linear extrapolation based on the first
764  * points of the original threshold.
765  *
766  * The tail is extended using the average of the last points of the
767  * original threshold.
768  */
769 
770  const float head_slope_multiplier = 2.0f;
771  const uint16_t head_slope_calculation_width = 2U;
772  const uint16_t tail_mean_calculation_width = 10U;
773  const uint16_t tail_mean_calculation_start = app->data_length - CFAR_MARGIN - tail_mean_calculation_width;
774 
775  const float head_base_offset = app->threshold[CFAR_MARGIN];
776  const uint16_t tail_offset = app->data_length - CFAR_MARGIN;
777 
778  /*
779  * Calculate head.
780  */
781 
782  float head_mean_slope = 0.0f;
783 
784  for (uint16_t i = 0; i < head_slope_calculation_width; i++)
785  {
786  uint16_t pos = CFAR_MARGIN + i;
787  head_mean_slope += app->threshold[pos + 1U] - app->threshold[pos];
788  }
789 
790  head_mean_slope /= (float)head_slope_calculation_width;
791  head_mean_slope *= head_slope_multiplier;
792 
793  /*
794  * Calculate tail.
795  */
796 
797  float tail_mean = 0.0f;
798 
799  for (int16_t i = 0; i < tail_mean_calculation_width; i++)
800  {
801  tail_mean += app->threshold[tail_mean_calculation_start + i];
802  }
803 
804  tail_mean /= (float)tail_mean_calculation_width;
805 
806  /*
807  * Extend the threshold at both ends.
808  */
809 
810  for (uint16_t i = 0; i < CFAR_MARGIN; i++)
811  {
812  app->threshold[i] = ((float)i - CFAR_MARGIN) * head_mean_slope + head_base_offset;
813  app->threshold[tail_offset + i] = tail_mean;
814  }
815 }

818 static float *get_zero_mean_time_series(acc_vibration_app_t *app, acc_vibration_config_t *config)

838 static float calculate_time_series_std(float *zm_time_series, uint16_t zm_time_series_length, float radians_to_displacement)
839 {
840  double time_series_mean = 0.0;
841 
842  for (uint16_t i = 0; i < zm_time_series_length; i++)
843  {
844  float zm_time_series_um = zm_time_series[i] * radians_to_displacement;
845  time_series_mean += (double)(zm_time_series_um * zm_time_series_um);
846  }
847 
848  time_series_mean /= (double)zm_time_series_length;
849 
850  return (float)sqrt(time_series_mean);
851 }
