199 bool i2c_ref_app_breathing_command(uint32_t command)
200 {
201  bool status = false;
202 
203  /* Make sure we do not have a race for i2c_app_command/i2c_app_status */
204  acc_integration_critical_section_enter();
205 
206  if (i2c_app_command == 0U)
207  {
208  /* Set Ready PIN to LOW while processing the command */
209  i2c_application_system_set_ready_pin(false);
210 
211  /* Set status BUSY bit */
212  i2c_app_status |= REF_APP_BREATHING_REG_APP_STATUS_FIELD_BUSY_MASK;
213  i2c_app_command = command;
214  status = true;
215  }
216 
217  acc_integration_critical_section_exit();
218  return status;
219 }

235 uint32_t i2c_ref_app_breathing_get_result(void)
236 {
237  uint32_t value = 0;
238 
239  /* Make sure we do not have a race for results */
240  acc_integration_critical_section_enter();
241 
242  if (result_ready)
243  {
244  value |= REF_APP_BREATHING_REG_BREATHING_RESULT_FIELD_RESULT_READY_MASK;
245  }
246 
247  if (result_ready_sticky)
248  {
249  value |= REF_APP_BREATHING_REG_BREATHING_RESULT_FIELD_RESULT_READY_STICKY_MASK;
250  result_ready_sticky = false;
251  }
252 
253  /* Add temperature */
254  uint32_t temp = (uint32_t)app_resources.result.presence_result.processing_result.temperature;
255 
256  temp = (temp << REF_APP_BREATHING_REG_BREATHING_RESULT_FIELD_TEMPERATURE_POS) &
257  REF_APP_BREATHING_REG_BREATHING_RESULT_FIELD_TEMPERATURE_MASK;
258  value |= temp;
259 
260  acc_integration_critical_section_exit();
261 
262  return value;
263 }

322 int acconeer_main(int argc, char *argv[])
323 {
324  (void)argc;
325  (void)argv;
326 
327  bool setup_status = true;
328  bool get_next_frame = false;
329 
330  printf("I2C Ref App Breathing\n");
331  printf("Acconeer software version %s\n", acc_version_get());
332 
333  const acc_hal_a121_t *hal = acc_hal_rss_integration_get_implementation();
334 
335  if (acc_rss_hal_register(hal))
336  {
337  app_status_set_bits(REF_APP_BREATHING_REG_APP_STATUS_FIELD_RSS_REGISTER_OK_MASK);
338  }
339  else
340  {
341  printf("ERROR: acc_rss_hal_register() failed\n\n");
342  app_status_set_bits(
343  REF_APP_BREATHING_REG_APP_STATUS_FIELD_APP_ERROR_MASK | REF_APP_BREATHING_REG_APP_STATUS_FIELD_RSS_REGISTER_ERROR_MASK);
344  setup_status = false;
345  }
346 
347  if (setup_status)
348  {
349  app_resources.config = ref_app_breathing_config_create();
350  if (app_resources.config != NULL)
351  {
352  /* Config is created, write default values to registers */
353  ref_app_breathing_reg_protocol_write_default();
354 
355  app_status_set_bits(REF_APP_BREATHING_REG_APP_STATUS_FIELD_CONFIG_CREATE_OK_MASK);
356  }
357  else
358  {
359  app_status_set_bits(
360  REF_APP_BREATHING_REG_APP_STATUS_FIELD_APP_ERROR_MASK |
361  REF_APP_BREATHING_REG_APP_STATUS_FIELD_CONFIG_CREATE_ERROR_MASK);
362  printf("ERROR: ref_app_breathing_config_create() failed\n\n");
363  setup_status = false;
364  }
365  }
366 
367  /* Turn the sensor on */
368  acc_hal_integration_sensor_supply_on(SENSOR_ID);
369 
370  if (setup_status)
371  {
372  /* Create sensor */
373  create_sensor(&app_resources);
374  }
375 
376  i2c_application_system_init();
377 
378  /* Setup i2c register protocol */
379  ref_app_breathing_reg_protocol_setup();
380 
381  while (true)
382  {
383  /* Handle Ref App Breathing */
384  if (ref_app_breathing_active &&
385  get_next_frame &&
386  !app_status_test_bits(REF_APP_BREATHING_REG_APP_STATUS_FIELD_APP_ERROR_MASK))
387  {
388  get_next_frame = false;
389  if (app_get_next(&app_resources))
390  {
391  print_breathing_result(&app_resources.result);
392  }
393  else
394  {
395  printf("ERROR: Could not get next result\n");
396  app_activate(&app_resources, false);
397  app_status_set_bits(REF_APP_BREATHING_REG_APP_STATUS_FIELD_APP_ERROR_MASK);
398  }
399  }
400 
401  /* Handle Commands */
402  uint32_t command = get_command();
403 
404  if (command == 0)
405  {
406  /* Try to set module in low power mode */
407  module_low_power();
408 
409  /* Test if a periodic wakeup was the wakeup reason */
410  get_next_frame = i2c_application_is_periodic_wakeup();
411  if (get_next_frame)
412  {
413  acc_integration_critical_section_enter();
414  uint32_t time_ms = acc_integration_get_time();
415  uint32_t diff_ms = time_ms - breathing_last_tick_ms;
416  breathing_last_tick_ms = time_ms;
417  breathing_frame_rate_mhz = (diff_ms > 0) ? (1000000 / diff_ms) : 0;
418 
419  acc_integration_critical_section_exit();
420  }
421 
422  continue;
423  }
424 
425  /* Special command, always handle reset module command, even if error has occured */
426  if (command == REF_APP_BREATHING_REG_COMMAND_ENUM_RESET_MODULE)
427  {
428  /* Reset system */
429  i2c_application_system_reset();
430  continue;
431  }
432 
433  if (app_status_test_bits(REF_APP_BREATHING_REG_APP_STATUS_FIELD_APP_ERROR_MASK))
434  {
435  /* Do not process commands after error state */
436  continue;
437  }
438 
439  /* Handle command */
440  command_handler(command);
441 
442  /* Command handler done, clear busy bit */
443  app_status_clr_bits(REF_APP_BREATHING_REG_APP_STATUS_FIELD_BUSY_MASK);
444 
445  /* Set Ready PIN to HIGH when command processing is done */
446  i2c_application_system_set_ready_pin(true);
447  }
448 
449  return EXIT_FAILURE;
450 }

459 {
460  /* Make sure we do not have a race for i2c_app_command */
461  acc_integration_critical_section_enter();
462 
463  uint32_t command = i2c_app_command;

468

472 
473 static void command_handler(uint32_t command)
474 {
475  switch (command)
476  {
477  case REF_APP_BREATHING_REG_COMMAND_ENUM_APPLY_CONFIGURATION:
478  if (!app_status_test_bits(REF_APP_BREATHING_REG_APP_STATUS_FIELD_CONFIG_APPLY_OK_MASK))
479  {
480  // Apply configuration
481  apply_app_config(&app_resources);
482  }
483 
484  break;
485  case REF_APP_BREATHING_REG_COMMAND_ENUM_START_APP:
486  if (!ref_app_breathing_active)
487  {
488  if (app_activate(&app_resources, true))
489  {
490  ref_app_breathing_active = true;
491  }
492  else
493  {
494  printf("ERROR: Could not start application\n");
495  app_status_set_bits(REF_APP_BREATHING_REG_APP_STATUS_FIELD_APP_ERROR_MASK);
496  }
497  }
498 
499  break;
500  case REF_APP_BREATHING_REG_COMMAND_ENUM_STOP_APP:
501  if (ref_app_breathing_active)
502  {
503  if (app_activate(&app_resources, false))
504  {
505  ref_app_breathing_active = false;
506  }
507  else
508  {
509  printf("ERROR: Could not stop application\n");
510  app_status_set_bits(REF_APP_BREATHING_REG_APP_STATUS_FIELD_APP_ERROR_MASK);
511  }
512  }
513 
514  break;
515  case REF_APP_BREATHING_REG_COMMAND_ENUM_ENABLE_UART_LOGS:
516  uart_logs_enabled = true;
517  uart_log("UART logs enabled\n");
518  break;
519  case REF_APP_BREATHING_REG_COMMAND_ENUM_DISABLE_UART_LOGS:
520  uart_log("UART logs disabled\n");
521  uart_logs_enabled = false;
522  break;
523  case REF_APP_BREATHING_REG_COMMAND_ENUM_LOG_CONFIGURATION:
524  // Print the configuration
525  acc_detector_presence_config_log(app_resources.config->presence_config);
526  break;
527  default:
528  printf("ERROR: Unknown command: %" PRIu32 "", command);
529  break;
530  }
531 }

598 static bool calibrate_sensor(ref_app_breathing_resources_t *resources)
599 {
600  app_status_clr_bits(
601  REF_APP_BREATHING_REG_APP_STATUS_FIELD_SENSOR_CALIBRATE_OK_MASK |
602  REF_APP_BREATHING_REG_APP_STATUS_FIELD_SENSOR_CALIBRATE_ERROR_MASK
603  );
604 
605  bool status = false;
606  bool cal_complete = false;
607  const uint16_t calibration_retries = 1U;
608 
609  // Random disturbances may cause the calibration to fail. At failure, retry at least once.
610  for (uint16_t i = 0; !status && (i <= calibration_retries); i++)
611  {
612  acc_hal_integration_sensor_enable(SENSOR_ID);
613 
614  do
615  {
616  status = acc_sensor_calibrate(resources->sensor,
617  &cal_complete,
618  &resources->sensor_cal_result,
619  resources->buffer,
620  resources->buffer_size);
621  if (status && !cal_complete)
622  {
623  status = acc_hal_integration_wait_for_sensor_interrupt(SENSOR_ID, SENSOR_TIMEOUT_MS);
624  }
625  } while (status && !cal_complete);
626 
627  /* Reset sensor after calibration by disabling it */
628  acc_hal_integration_sensor_disable(SENSOR_ID);
629  }
630 
631  if (status)
632  {
633  app_status_set_bits(REF_APP_BREATHING_REG_APP_STATUS_FIELD_SENSOR_CALIBRATE_OK_MASK);
634  }
635  else
636  {
637  app_status_set_bits(REF_APP_BREATHING_REG_APP_STATUS_FIELD_SENSOR_CALIBRATE_ERROR_MASK);
638  printf("ERROR: acc_sensor_calibrate() failed\n");
639  }
640 
641  return status;
642 }

645 static void apply_app_config(ref_app_breathing_resources_t *resources)
646 {
647  bool status = true;
648 
649  acc_detector_presence_config_frame_rate_app_driven_set(resources->config->presence_config, true);
650 
651  /* Always use DEEP_SLEEP for inter_frame_idle_state to save power */
652  acc_detector_presence_config_inter_frame_idle_state_set(resources->config->presence_config, ACC_CONFIG_IDLE_STATE_DEEP_SLEEP);
653 
654  resources->handle = ref_app_breathing_create(resources->config);
655  if (resources->handle != NULL)
656  {
657  app_status_set_bits(REF_APP_BREATHING_REG_APP_STATUS_FIELD_APP_CREATE_OK_MASK);
658  }
659  else
660  {
661  app_status_set_bits(
662  REF_APP_BREATHING_REG_APP_STATUS_FIELD_APP_ERROR_MASK |
663  REF_APP_BREATHING_REG_APP_STATUS_FIELD_APP_CREATE_ERROR_MASK
664  );
665  printf("ERROR: ref_app_breathing_create() failed\n");
666  status = false;
667  }
668 
669  if (status)
670  {
671  if (ref_app_breathing_get_buffer_size(resources->handle, &(resources->buffer_size)))
672  {
673  app_status_set_bits(REF_APP_BREATHING_REG_APP_STATUS_FIELD_APP_BUFFER_OK_MASK);
674  }
675  else
676  {
677  app_status_set_bits(
678  REF_APP_BREATHING_REG_APP_STATUS_FIELD_APP_ERROR_MASK |
679  REF_APP_BREATHING_REG_APP_STATUS_FIELD_APP_BUFFER_ERROR_MASK
680  );
681  printf("ERROR: ref_app_breathing_get_buffer_size() failed\n");
682  status = false;
683  }
684  }
685 
686  if (status)
687  {
688  resources->buffer = acc_integration_mem_alloc(resources->buffer_size);
689  if (resources->buffer != NULL)
690  {
691  app_status_set_bits(REF_APP_BREATHING_REG_APP_STATUS_FIELD_SENSOR_BUFFER_OK_MASK);
692  }
693  else
694  {
695  app_status_set_bits(
696  REF_APP_BREATHING_REG_APP_STATUS_FIELD_APP_ERROR_MASK |
697  REF_APP_BREATHING_REG_APP_STATUS_FIELD_SENSOR_BUFFER_ERROR_MASK
698  );
699  printf("ERROR: sensor buffer allocation failed\n");
700  status = false;
701  }
702  }
703 
704  if (status)
705  {
706  status = calibrate_sensor(&app_resources);
707  }
708 
709  if (status)
710  {
711  app_status_set_bits(REF_APP_BREATHING_REG_APP_STATUS_FIELD_CONFIG_APPLY_OK_MASK);
712  }
713  else
714  {
715  printf("ERROR: apply application config failed\n");
716  app_status_set_bits(
717  REF_APP_BREATHING_REG_APP_STATUS_FIELD_APP_ERROR_MASK |
718  REF_APP_BREATHING_REG_APP_STATUS_FIELD_CONFIG_APPLY_ERROR_MASK
719  );
720  }
721 }
