39 void ref_app_breathing_reg_read_protocol_status(uint32_t *value)
40 {
41  uint32_t error_flags = get_protocol_error_flags();
42  uint32_t protocol_status = 0;
43 
44  if ((error_flags & ACC_REG_ERROR_FLAG_PROTOCOL_STATE_ERROR) != 0)
45  {
46  protocol_status |= REF_APP_BREATHING_REG_PROTOCOL_STATUS_FIELD_PROTOCOL_STATE_ERROR_MASK;
47  }
48 
49  if ((error_flags & ACC_REG_ERROR_FLAG_PACKET_LENGTH_ERROR) != 0)
50  {
51  protocol_status |= REF_APP_BREATHING_REG_PROTOCOL_STATUS_FIELD_PACKET_LENGTH_ERROR_MASK;
52  }
53 
54  if ((error_flags & ACC_REG_ERROR_FLAG_ADDRESS_ERROR) != 0)
55  {
56  protocol_status |= REF_APP_BREATHING_REG_PROTOCOL_STATUS_FIELD_ADDRESS_ERROR_MASK;
57  }
58 
59  if ((error_flags & ACC_REG_ERROR_FLAG_WRITE_FAILED) != 0)
60  {
61  protocol_status |= REF_APP_BREATHING_REG_PROTOCOL_STATUS_FIELD_WRITE_FAILED_MASK;
62  }
63 
64  if ((error_flags & ACC_REG_ERROR_FLAG_WRITE_TO_READ_ONLY) != 0)
65  {
66  protocol_status |= REF_APP_BREATHING_REG_PROTOCOL_STATUS_FIELD_WRITE_TO_READ_ONLY_MASK;
67  }
68 
69  *value = protocol_status;
70 }
