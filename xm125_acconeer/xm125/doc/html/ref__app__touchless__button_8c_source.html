216 int acconeer_main(int argc, char *argv[])
217 {
218  (void)argc;
219  (void)argv;
220 
221  acc_touchless_button_handle_t handle;
222  handle.sensor = NULL;
223  handle.processing = NULL;
224  handle.buffer = NULL;
225  handle.double_buffer_filter_buffer = NULL;
226  handle.frame_variance = NULL;
227  handle.arg_norm = NULL;
228  handle.ampl_mean = NULL;
229  handle.ampl_std = NULL;
230  handle.phase_mean = NULL;
231  handle.phase_std = NULL;
232  handle.dynamic_background = NULL;
233  handle.dynamic_background_guard = NULL;
234  handle.threshold_check_count = NULL;
235 
236  printf("Acconeer software version %s\n", acc_version_get());
237 
238  const acc_hal_a121_t *hal = acc_hal_rss_integration_get_implementation();
239 
240  if (!acc_rss_hal_register(hal))
241  {
242  return EXIT_FAILURE;
243  }
244 
245  handle.config.sensor_config = acc_config_create();
246  if (handle.config.sensor_config == NULL)
247  {
248  printf("acc_config_create() failed\n");
249  cleanup(&handle);
250  return EXIT_FAILURE;
251  }
252 
253  set_config(&handle.config, DEFAULT_PRESET);
254 
255  if (!validate_config(&handle.config))
256  {
257  cleanup(&handle);
258  return EXIT_FAILURE;
259  }
260 
261  config_log(&handle.config);
262 
263  if (!init_handle(&handle))
264  {
265  printf("Failed to init handle\n");
266  cleanup(&handle);
267  return EXIT_FAILURE;
268  }
269 
270  if (!init_sensor(&handle))
271  {
272  printf("Failed to init sensor\n");
273  cleanup(&handle);
274  return EXIT_FAILURE;
275  }
276 
277  while (true)
278  {
279  if (!measure(&handle))
280  {
281  printf("Failed to measure\n");
282  cleanup(&handle);
283  return EXIT_FAILURE;
284  }
285 
286  if (handle.proc_result.data_saturated)
287  {
288  printf("Data saturated. The detector result is not reliable.\n");
289  }
290 
291  if (handle.proc_result.frame_delayed)
292  {
293  printf("Frame delayed. Could not read data fast enough.\n");
294  printf("Try lowering the frame rate or call 'acc_sensor_read' more frequently.\n");
295  }
296 
297  if (handle.proc_result.calibration_needed)
298  {
299  printf("The current calibration is not valid for the current temperature.\n");
300  printf("The sensor needs to be re-calibrated.\n");
301 
302  if (!do_sensor_calibration_and_prepare(&handle))
303  {
304  printf("do_sensor_calibration_and_prepare() failed\n");
305  acc_sensor_status(handle.sensor);
306  cleanup(&handle);
307  return EXIT_FAILURE;
308  }
309 
310  reset_background(&handle);
311 
312  printf("The sensor was successfully re-calibrated.\n");
313  }
314  else
315  {
316  acc_touchless_button_result_t result = { 0 };
317 
318  if (!process(&handle, &result))
319  {
320  printf("Failed to process sensor result\n");
321  cleanup(&handle);
322  return EXIT_FAILURE;
323  }
324 
325  print_result(&result);
326  }
327  }
328 
329  cleanup(&handle);
330 
331  printf("Application finished OK\n");
332 
333  return EXIT_SUCCESS;
334 }

495 static bool validate_config(acc_touchless_button_config_t *config)
496 {
497  bool status = true;
498 
499  if (acc_config_num_subsweeps_get(config->sensor_config) != 2U &&
500  config->measurement_type == ACC_TOUCHLESS_BUTTON_CLOSE_AND_FAR_RANGE)
501  {
502  printf("Number of subsweeps must be 2 for 'close and far'\n");
503  status = false;
504  }
505 
506  if (acc_config_num_subsweeps_get(config->sensor_config) != 1U &&
507  (config->measurement_type == ACC_TOUCHLESS_BUTTON_CLOSE_RANGE || config->measurement_type == ACC_TOUCHLESS_BUTTON_FAR_RANGE))
508  {
509  printf("Number of subsweeps must be 1 for 'close' and 'far'\n");
510  status = false;
511  }
512 
513  float sweep_rate = acc_config_sweep_rate_get(config->sensor_config);
514  uint16_t spf = acc_config_sweeps_per_frame_get(config->sensor_config);
515 
516  if (sweep_rate == 0.0f)
517  {
518  printf("Sweep rate must be set\n");
519  status = false;
520  }
521  else
522  {
523  if (spf > (config->calibration_duration_s * sweep_rate))
524  {
525  float calibration_limit = (float)spf / sweep_rate;
526 
527  printf("Calibration duration must be at least %" PRIfloat " s\n", ACC_LOG_FLOAT_TO_INTEGER(calibration_limit));
528  status = false;
529  }
530  }
531 
532  return status;
533 }

577 
578  printf("measurement type: %s\n", measurement_type_str);
579 
580  printf("Sensor config\n");
581  acc_config_log(config->sensor_config);

585 static bool init_handle(acc_touchless_button_handle_t *handle)
586 {
587  bool status = true;
588 
589  if (!acc_rss_get_buffer_size(handle->config.sensor_config, &handle->buffer_size))
590  {
591  printf("acc_rss_get_buffer_size() failed\n");
592  status = false;
593  }
594 
595  if (status)
596  {
597  handle->processing = acc_processing_create(handle->config.sensor_config, &handle->proc_metadata);
598  handle->buffer = acc_integration_mem_alloc(handle->buffer_size);
599 
600  uint16_t spf = acc_config_sweeps_per_frame_get(handle->config.sensor_config);
601  float sweep_rate = acc_config_sweep_rate_get(handle->config.sensor_config);
602  float frame_rate = sweep_rate / (float)spf;
603 
604  handle->cal_interval_frames = (uint16_t)((frame_rate * handle->config.calibration_interval_s) + 0.5f);
605  handle->cal_sweeps = (uint16_t)((sweep_rate * handle->config.calibration_duration_s) + 0.5f);
606  uint16_t num_points = handle->proc_metadata.sweep_data_length;
607 
608  handle->double_buffer_filter_buffer =
609  acc_integration_mem_alloc((spf - 2U) * sizeof(*handle->double_buffer_filter_buffer));
610  handle->frame_variance =
611  acc_integration_mem_alloc(handle->proc_metadata.frame_data_length * sizeof(*handle->frame_variance));
612  handle->arg_norm = acc_integration_mem_alloc(handle->proc_metadata.sweep_data_length * sizeof(*handle->arg_norm));
613  handle->ampl_mean = acc_integration_mem_alloc(handle->proc_metadata.sweep_data_length * sizeof(*handle->ampl_mean));
614  handle->ampl_std = acc_integration_mem_alloc(handle->proc_metadata.sweep_data_length * sizeof(*handle->ampl_std));
615  handle->phase_mean = acc_integration_mem_alloc(handle->proc_metadata.sweep_data_length * sizeof(*handle->phase_mean));
616  handle->phase_std = acc_integration_mem_alloc(handle->proc_metadata.sweep_data_length * sizeof(*handle->phase_std));
617  handle->dynamic_background = acc_integration_mem_alloc(handle->cal_sweeps * num_points * sizeof(*handle->dynamic_background));
618  handle->dynamic_background_guard =
619  acc_integration_mem_alloc(handle->proc_metadata.frame_data_length * sizeof(*handle->dynamic_background_guard));
620  handle->threshold_check_count =
621  acc_integration_mem_alloc(handle->proc_metadata.sweep_data_length * sizeof(*handle->threshold_check_count));
622 
623  status =
624  (handle->processing != NULL && handle->buffer != NULL && handle->double_buffer_filter_buffer != NULL &&
625  handle->frame_variance != NULL && handle->arg_norm != NULL && handle->ampl_mean != NULL &&
626  handle->ampl_std != NULL && handle->phase_mean != NULL && handle->phase_std != NULL &&
627  handle->dynamic_background != NULL && handle->dynamic_background_guard != NULL &&
628  handle->threshold_check_count != NULL);

768 static void calc_variance(acc_touchless_button_handle_t *handle)
769 {
770  uint16_t sweep_data_length = handle->proc_metadata.sweep_data_length;
771 
772  acc_algorithm_mean_matrix_i16_complex(handle->dynamic_background, handle->cal_sweeps, sweep_data_length, handle->arg_norm, 0U);
773 
774  acc_algorithm_conj_f32(handle->arg_norm, sweep_data_length);
775 
776  acc_algorithm_normalize_f32_complex(handle->arg_norm, sweep_data_length);
777 
778  for (uint16_t c = 0U; c < sweep_data_length; c++)
779  {
780  float abs_mean = 0.0f;
781  float abs_sq_term = 0.0f;
782  float phase_mean = 0.0f;
783  float phase_sq_term = 0.0f;
784  for (uint16_t r = 0U; r < handle->cal_sweeps; r++)
785  {
786  float ac = handle->dynamic_background[r * sweep_data_length + c].real * crealf(handle->arg_norm[c]);
787  float bd = handle->dynamic_background[r * sweep_data_length + c].imag * cimagf(handle->arg_norm[c]);
788  float ad = handle->dynamic_background[r * sweep_data_length + c].real * cimagf(handle->arg_norm[c]);
789  float bc = handle->dynamic_background[r * sweep_data_length + c].imag * crealf(handle->arg_norm[c]);
790  float complex element = (ac - bd) + I * (ad + bc);
791 
792  float delta = cabsf(element) - abs_mean;
793  uint16_t div = (r == 0U) ? 1U : r;
794  abs_mean += (delta / (float)div);
795  abs_sq_term += delta * (cabsf(element) - abs_mean);
796 
797  delta = cargf(element) - phase_mean;
798  phase_mean += (delta / (float)div);
799  phase_sq_term += delta * (cargf(element) - phase_mean);
800  }
801 
802  handle->ampl_mean[c] = abs_mean;
803  handle->ampl_std[c] = sqrtf(abs_sq_term / handle->cal_sweeps);
804  handle->phase_mean[c] = phase_mean;
805  handle->phase_std[c] = sqrtf(phase_sq_term / handle->cal_sweeps);
806  }
807 
808  uint16_t spf = acc_config_sweeps_per_frame_get(handle->config.sensor_config);
809 
810  for (uint16_t r = 0U; r < spf; r++)
811  {
812  for (uint16_t c = 0U; c < sweep_data_length; c++)
813  {
814  float ac = handle->proc_result.frame[r * sweep_data_length + c].real * crealf(handle->arg_norm[c]);
815  float bd = handle->proc_result.frame[r * sweep_data_length + c].imag * cimagf(handle->arg_norm[c]);
816  float ad = handle->proc_result.frame[r * sweep_data_length + c].real * cimagf(handle->arg_norm[c]);
817  float bc = handle->proc_result.frame[r * sweep_data_length + c].imag * crealf(handle->arg_norm[c]);
818  float complex element = (ac - bd) + I * (ad + bc);
819  float a = (cabsf(element) - handle->ampl_mean[c]) / handle->ampl_std[c];
820  float b = (cargf(element) - handle->phase_mean[c]) / handle->phase_std[c];
821 
822  handle->frame_variance[r * sweep_data_length + c] = sqrtf(powf(a, 2.0f) + powf(b, 2.0f));
823  }
824  }
825 }

868 static bool process(acc_touchless_button_handle_t *handle, acc_touchless_button_result_t *result)
869 {

871  {
872  acc_algorithm_double_buffering_frame_filter(handle->proc_result.frame, acc_config_sweeps_per_frame_get(
873  handle->config.sensor_config),
874  acc_config_num_points_get(handle->config.sensor_config),
875  handle->double_buffer_filter_buffer);
876  }
877 
878  memset(handle->frame_variance, 0, handle->proc_metadata.frame_data_length * sizeof(*handle->frame_variance));
879 
880  if (handle->frames_since_last_cal > handle->cal_interval_frames)
881  {
882  reset_background(handle);
883  }
884  else if (handle->rows_in_dynamic_background == handle->cal_sweeps)
885  {
886  calc_variance(handle);
887  }
888 
889  uint16_t threshold_check_offset = 0U;
890 
891  if (handle->run_close)
892  {
893  acc_algorithm_count_points_above_threshold(handle->frame_variance, acc_config_sweeps_per_frame_get(
894  handle->config.sensor_config), handle->proc_metadata.sweep_data_length, handle->close_threshold, handle->threshold_check_count, threshold_check_offset, handle->close_num_points,
895  1U);
896 
897  threshold_check_offset = handle->close_num_points;
898  }
899 
900  if (handle->run_far)
901  {
902  acc_algorithm_count_points_above_threshold(handle->frame_variance, acc_config_sweeps_per_frame_get(
903  handle->config.sensor_config), handle->proc_metadata.sweep_data_length, handle->far_threshold, handle->threshold_check_count, threshold_check_offset, handle->far_num_points,
904  1U);
905  }
906 
907  if (check_count(handle->threshold_check_count, 0U, handle->proc_metadata.sweep_data_length))
908  {
909  uint16_t close_offset = 0U;
910  uint16_t far_offset = handle->close_num_points;
911 
912  if (handle->run_close)
913  {
914  if (check_count(handle->threshold_check_count, close_offset, handle->close_num_points))
915  {
916  handle->close_signal++;
917  handle->close_non_signal = 0U;
918  }
919  else
920  {
921  handle->close_signal = 0U;
922  handle->close_non_signal++;
923  }
924  }
925 
926  if (handle->run_far)
927  {
928  if (check_count(handle->threshold_check_count, far_offset, handle->far_num_points))
929  {
930  handle->far_signal++;
931  handle->far_non_signal = 0U;
932  }
933  else
934  {
935  handle->far_signal = 0U;
936  handle->far_non_signal++;
937  }
938  }
939 
940  handle->update_background = false;
941  memset(handle->dynamic_background_guard, 0, handle->proc_metadata.frame_data_length * sizeof(*handle->dynamic_background_guard));
942  handle->frames_since_last_cal++;
943  }
944  else
945  {
946  handle->close_signal = 0U;
947  handle->far_signal = 0U;
948  handle->close_non_signal++;
949  handle->far_non_signal++;
950 
951  if (handle->update_background)
952  {
953  update_background(handle);
954  }
955 
956  memcpy(handle->dynamic_background_guard, handle->proc_result.frame,
957  handle->proc_metadata.frame_data_length * sizeof(*handle->dynamic_background_guard));
958  handle->update_background = true;
959  }
960 
961  memset(result, 0, sizeof(*result));
962 
963  result->close_result = ACC_TOUCHLESS_BUTTON_RANGE_NOT_USED;
964  result->far_result = ACC_TOUCHLESS_BUTTON_RANGE_NOT_USED;
965 
966  if (handle->run_close)
967  {
968  handle->close_detection = get_detection(handle->close_detection, handle->close_signal, handle->close_non_signal,
969  handle->config.patience_close);
970  result->close_result = handle->close_detection ? ACC_TOUCHLESS_BUTTON_RANGE_DETECTION : ACC_TOUCHLESS_BUTTON_RANGE_NO_DETECTION;
971  }
972 
973  if (handle->run_far)
974  {
975  handle->far_detection = get_detection(handle->far_detection, handle->far_signal, handle->far_non_signal, handle->config.patience_far);
976  result->far_result = handle->far_detection ? ACC_TOUCHLESS_BUTTON_RANGE_DETECTION : ACC_TOUCHLESS_BUTTON_RANGE_NO_DETECTION;
977  }
978 
979  return true;
980 }
