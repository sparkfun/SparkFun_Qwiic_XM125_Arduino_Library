67 int acconeer_main(int argc, char *argv[])
68 {
69  (void)argc;
70  (void)argv;
71  ref_app_breathing_config_t *config = NULL;
72  ref_app_breathing_handle_t *handle = NULL;
73  acc_sensor_t *sensor = NULL;
74  acc_cal_result_t sensor_cal_result;
75  void *buffer = NULL;
76  uint32_t buffer_size = 0U;
77  ref_app_breathing_app_state_t prev_app_state = (ref_app_breathing_app_state_t)0U;
78 
79  printf("Acconeer software version %s\n", acc_version_get());
80 
81  const acc_hal_a121_t *hal = acc_hal_rss_integration_get_implementation();
82 
83  if (!acc_rss_hal_register(hal))
84  {
85  return EXIT_FAILURE;
86  }
87 
88  config = ref_app_breathing_config_create();
89 
90  if (config == NULL)
91  {
92  printf("Failed to create config\n");
93  cleanup(handle, config, sensor, buffer);
94  return EXIT_FAILURE;
95  }
96 
97  set_config(config, DEFAULT_PRESET_CONFIG);
98 
99  handle = ref_app_breathing_create(config);
100 
101  if (handle == NULL)
102  {
103  printf("Failed to create handle\n");
104  cleanup(handle, config, sensor, buffer);
105  return EXIT_FAILURE;
106  }
107 
108  if (!ref_app_breathing_get_buffer_size(handle, &buffer_size))
109  {
110  printf("ref_app_breathing_get_buffer_size() failed\n");
111  cleanup(handle, config, sensor, buffer);
112  return EXIT_FAILURE;
113  }
114 
115  buffer = acc_integration_mem_alloc(buffer_size);
116  if (buffer == NULL)
117  {
118  printf("Failed to allocate buffer\n");
119  cleanup(handle, config, sensor, buffer);
120  return EXIT_FAILURE;
121  }
122 
123  float frame_rate = acc_detector_presence_config_frame_rate_get(config->presence_config);
124  uint32_t sleep_time_ms = (uint32_t)(1000.0f / frame_rate);
125  acc_integration_set_periodic_wakeup(sleep_time_ms);
126 
127  acc_hal_integration_sensor_supply_on(SENSOR_ID);
128  acc_hal_integration_sensor_enable(SENSOR_ID);
129 
130  sensor = acc_sensor_create(SENSOR_ID);
131  if (sensor == NULL)
132  {
133  printf("acc_sensor_create() failed\n");
134  cleanup(handle, config, sensor, buffer);
135  return EXIT_FAILURE;
136  }
137 
138  if (!sensor_calibration(sensor, &sensor_cal_result, buffer, buffer_size))
139  {
140  printf("Sensor calibration failed\n");
141  cleanup(handle, config, sensor, buffer);
142  return EXIT_FAILURE;
143  }
144 
145  if (!ref_app_breathing_prepare(handle, config, sensor, &sensor_cal_result, buffer, buffer_size))
146  {
147  printf("ref_app_breathing_prepare() failed\n");
148  cleanup(handle, config, sensor, buffer);
149  return EXIT_FAILURE;
150  }
151 
152  ref_app_breathing_result_t result = { 0 };
153 
154  if (!enter_hibernate(sensor))
155  {
156  printf("enter_hibernate failed\n");
157  cleanup(handle, config, sensor, buffer);
158  return EXIT_FAILURE;
159  }
160 
161  while (true)
162  {
163  if (!measure(sensor, buffer, buffer_size))
164  {
165  cleanup(handle, config, sensor, buffer);
166  return EXIT_FAILURE;
167  }
168 
169  if (!ref_app_breathing_process(handle, buffer, &result))
170  {
171  printf("ref_app_breathing_process() failed\n");
172  cleanup(handle, config, sensor, buffer);
173  return EXIT_FAILURE;
174  }
175 
176  if (!handle_indications(handle, config, sensor, &sensor_cal_result, buffer, buffer_size, &result.presence_result))
177  {
178  cleanup(handle, config, sensor, buffer);
179  return EXIT_FAILURE;
180  }
181 
182  if (!result.presence_result.processing_result.calibration_needed)
183  {
184  print_result(&result, prev_app_state);
185  prev_app_state = result.app_state;
186  }
187 
188  acc_integration_sleep_until_periodic_wakeup();
189  }
190 
191  cleanup(handle, config, sensor, buffer);
192 
193  printf("Application finished OK\n");
194 
195  return EXIT_SUCCESS;
196 }

223 static void set_config(ref_app_breathing_config_t *config, breathing_preset_t preset)
224 {
225  acc_detector_presence_config_t *presence_config = config->presence_config;
226 
227  switch (preset)
228  {
229  case BREATHING_PRESET_NONE:
230  // Add custom configuration of the reference application here
231  break;
232  case BREATHING_PRESET_SITTING:
233  acc_detector_presence_config_end_set(presence_config, 1.5f);
234  acc_detector_presence_config_intra_detection_threshold_set(presence_config, 6.0f);
235  break;
236  case BREATHING_PRESET_INFANT:
237  acc_detector_presence_config_end_set(presence_config, 1.0f);
238  acc_detector_presence_config_intra_detection_threshold_set(presence_config, 4.0f);
239  break;
240  }
241 
242  acc_detector_presence_config_frame_rate_app_driven_set(presence_config, true);
243 }

246 static bool sensor_calibration(acc_sensor_t *sensor, acc_cal_result_t *sensor_cal_result, void *buffer, uint32_t buffer_size)
247 {
248  bool status = false;
249  bool cal_complete = false;
250  const uint16_t calibration_retries = 1U;
251 
252  // Random disturbances may cause the calibration to fail. At failure, retry at least once.
253  for (uint16_t i = 0; !status && (i <= calibration_retries); i++)
254  {
255  // Reset sensor before calibration by disabling/enabling it
256  acc_hal_integration_sensor_disable(SENSOR_ID);
257  acc_hal_integration_sensor_enable(SENSOR_ID);
258 
259  do
260  {
261  status = acc_sensor_calibrate(sensor, &cal_complete, sensor_cal_result, buffer, buffer_size);
262 
263  if (status && !cal_complete)
264  {
265  status = acc_hal_integration_wait_for_sensor_interrupt(SENSOR_ID, SENSOR_TIMEOUT_MS);
266  }
267  } while (status && !cal_complete);
268  }
269 
270  if (status)
271  {
272  /* Reset sensor after calibration by disabling/enabling it */
273  acc_hal_integration_sensor_disable(SENSOR_ID);
274  acc_hal_integration_sensor_enable(SENSOR_ID);
275  }
276  else
277  {
278  printf("acc_sensor_calibrate() failed\n");
279  acc_sensor_status(sensor);
280  }
281 
282  return status;
283 }

286 static bool measure(acc_sensor_t *sensor, void *buffer, uint32_t buffer_size)
287 {
288  if (!exit_hibernate(sensor))
289  {
290  printf("exit_hibernate failed\n");

292  }
293 
294  if (!acc_sensor_measure(sensor))
295  {
296  printf("acc_sensor_measure failed\n");
297  acc_sensor_status(sensor);
298  return false;
299  }
300 
301  if (!acc_hal_integration_wait_for_sensor_interrupt(SENSOR_ID, SENSOR_TIMEOUT_MS))
302  {
303  printf("Sensor interrupt timeout\n");
304  acc_sensor_status(sensor);
305  return false;
306  }
307 
308  if (!acc_sensor_read(sensor, buffer, buffer_size))
309  {
310  printf("acc_sensor_read() failed\n");
311  acc_sensor_status(sensor);
312  return false;
313  }
314 
315  if (!enter_hibernate(sensor))
316  {
317  printf("enter_hibernate failed\n");
318  return EXIT_FAILURE;
319  }
320 
321  return true;
322 }

395 static bool handle_indications(ref_app_breathing_handle_t *handle, ref_app_breathing_config_t *config, acc_sensor_t *sensor,
396  acc_cal_result_t *sensor_cal_result, void *buffer, uint32_t buffer_size,
397  acc_detector_presence_result_t *presence_result)
398 {
399  if (presence_result->processing_result.data_saturated)
400  {
401  printf("Data saturated. The detector result is not reliable.\n");
402  }
403 
404  if (presence_result->processing_result.frame_delayed)
405  {
406  printf("Frame delayed. Could not read data fast enough.\n");
407  printf("Try lowering the frame rate or call 'acc_sensor_read' more frequently.\n");
408  }
409 
410  if (presence_result->processing_result.calibration_needed)
411  {
412  printf("Sensor recalibration needed ... \n");
413 
414  if (!exit_hibernate(sensor))
415  {
416  printf("exit_hibernate failed\n");
417  return EXIT_FAILURE;
418  }
419 
420  if (!sensor_calibration(sensor, sensor_cal_result, buffer, buffer_size))
421  {
422  printf("Sensor calibration failed\n");
423  return false;
424  }
425 
426  printf("Sensor recalibration done!\n");
427 
428  // Before measuring again, the sensor needs to be prepared through the detector
429  if (!ref_app_breathing_prepare(handle, config, sensor, sensor_cal_result, buffer, buffer_size))
430  {
431  printf("acc_detector_presence_prepare() failed\n");
432  return false;
433  }
434 
435  if (!enter_hibernate(sensor))
436  {
437  printf("enter_hibernate failed\n");
438  return EXIT_FAILURE;
439  }
440  }
441 
442  return true;
443 }
