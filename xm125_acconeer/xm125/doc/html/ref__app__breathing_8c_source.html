108 ref_app_breathing_config_t *ref_app_breathing_config_create(void)
109 {
110  ref_app_breathing_config_t *config = acc_integration_mem_alloc(sizeof(*config));
111 
112  if (config != NULL)
113  {
114  config->presence_config = acc_detector_presence_config_create();
115 
116  if (config->presence_config == NULL)
117  {
118  ref_app_breathing_config_destroy(config);
119  return NULL;
120  }
121 
122  config->time_series_length_s = 20U;
123  config->lowest_breathing_rate = 6U;
124  config->highest_breathing_rate = 60U;
125  config->num_dists_to_analyze = 3U;
126  config->use_presence_processor = true;
127  config->distance_determination_duration_s = 5U;
128 
129  acc_detector_presence_config_t *presence_config = config->presence_config;
130 
131  acc_detector_presence_config_start_set(presence_config, 0.3f);
132  acc_detector_presence_config_end_set(presence_config, 1.5f);
133  acc_detector_presence_config_hwaas_set(presence_config, 32U);
134  acc_detector_presence_config_frame_rate_set(presence_config, 10.0f);
135  acc_detector_presence_config_sweeps_per_frame_set(presence_config, 16U);
136  acc_detector_presence_config_auto_profile_set(presence_config, false);
137  acc_detector_presence_config_profile_set(presence_config, ACC_CONFIG_PROFILE_3);
138  acc_detector_presence_config_auto_step_length_set(presence_config, false);
139  acc_detector_presence_config_step_length_set(presence_config, 24U);
140  acc_detector_presence_config_inter_frame_presence_timeout_set(presence_config, 0U);
141  acc_detector_presence_config_inter_frame_fast_cutoff_set(presence_config, 20.0f);
142  acc_detector_presence_config_intra_detection_threshold_set(presence_config, 6.0f);
143  acc_detector_presence_config_intra_output_time_const_set(presence_config, 0.5f);
144  acc_detector_presence_config_inter_output_time_const_set(presence_config, 5.0f);
145  }
146 
147  return config;

165 ref_app_breathing_handle_t *ref_app_breathing_create(ref_app_breathing_config_t *config)
166 {
167  if (!validate_config(config))
168  {
169  return NULL;
170  }
171 
172  ref_app_breathing_handle_t *handle = acc_integration_mem_alloc(sizeof(*handle));
173 
174  if (handle != NULL)
175  {
176  handle->frame_rate = acc_detector_presence_config_frame_rate_get(config->presence_config);
177  handle->sweeps_per_frame = acc_detector_presence_config_sweeps_per_frame_get(config->presence_config);
178  handle->intra_detection_threshold = acc_detector_presence_config_intra_detection_threshold_get(config->presence_config);
179 
180  acc_detector_presence_metadata_t presence_metadata;
181 
182  handle->presence_handle = acc_detector_presence_create(config->presence_config, &presence_metadata);
183 
184  if (handle->presence_handle == NULL)
185  {
186  ref_app_breathing_destroy(handle);
187  return NULL;
188  }
189 
190  handle->start_m = presence_metadata.start_m;
191  handle->step_length_m = presence_metadata.step_length_m;
192  handle->num_points = presence_metadata.num_points;
193 
194  handle->lowest_freq = (float)config->lowest_breathing_rate / 60.0f;
195  handle->highest_freq = (float)config->highest_breathing_rate / 60.0f;
196  handle->use_presence_processor = config->use_presence_processor;
197  handle->time_series_length_s = config->time_series_length_s;
198  handle->distance_determination_count = config->distance_determination_duration_s * handle->frame_rate;
199  handle->num_points_to_analyze_half_width = config->num_dists_to_analyze / 2U;
200  handle->num_points_to_analyze = config->use_presence_processor ? handle->num_points_to_analyze_half_width * 2U +
201  1U : handle->num_points;
202 
203  handle->time_series_length = handle->time_series_length_s * handle->frame_rate;
204  handle->padded_time_series_length_shift = 0U;
205  handle->padded_time_series_length = 1U << handle->padded_time_series_length_shift;
206 
207  while (handle->padded_time_series_length < handle->time_series_length)
208  {
209  handle->padded_time_series_length_shift++;
210  handle->padded_time_series_length = 1U << handle->padded_time_series_length_shift;
211  }
212 
213  handle->rfft_output_length = (handle->padded_time_series_length / 2U) + 1U;
214  handle->distance_determination_counter = 0U;
215  handle->presence_init = false;
216  handle->presence_distance = 0.0f;
217  handle->base_presence_dist = false;
218  handle->base_presence_distance = 0.0f;
219  handle->presence_distance_threshold = acc_algorithm_get_fwhm(acc_detector_presence_config_profile_get(config->presence_config)) *
220  2.0f;
221  handle->first = true;
222  handle->init_count = 0U;
223  handle->count = 0U;
224  handle->initialized = false;
225  handle->presence_sf = acc_algorithm_exp_smoothing_coefficient(handle->frame_rate,
226  (float)config->distance_determination_duration_s / 4.0f);
227  handle->breathing_sf = acc_algorithm_exp_smoothing_coefficient(handle->frame_rate, handle->time_series_length_s / 2.0f);
228 
229  handle->app_state = REF_APP_BREATHING_APP_STATE_INIT;
230  handle->prev_app_state = REF_APP_BREATHING_APP_STATE_INIT;
231 
232  handle->count_limit = handle->time_series_length / 2U;
233 
234  acc_algorithm_butter_lowpass(handle->lowest_freq, handle->frame_rate, handle->b_static, handle->a_static);
235  acc_algorithm_butter_bandpass(handle->lowest_freq, handle->highest_freq, handle->frame_rate,
236  handle->b_angle, handle->a_angle);
237 
238  handle->mean_sweep = acc_integration_mem_alloc(handle->num_points_to_analyze * sizeof(*handle->mean_sweep));
239  handle->filt_sparse_iq = acc_integration_mem_alloc(handle->num_points_to_analyze * sizeof(*handle->filt_sparse_iq));
240  handle->sparse_iq_buffer =
241  acc_integration_mem_alloc(B_STATIC_LENGTH * handle->num_points_to_analyze * sizeof(*handle->sparse_iq_buffer));
242  handle->filt_sparse_iq_buffer = acc_integration_mem_alloc(A_STATIC_LENGTH * handle->num_points_to_analyze *
243  sizeof(*handle->filt_sparse_iq_buffer));
244  handle->angle = acc_integration_mem_alloc(handle->num_points_to_analyze * sizeof(*handle->angle));
245  handle->prev_angle = acc_integration_mem_alloc(handle->num_points_to_analyze * sizeof(*handle->prev_angle));
246  handle->lp_filt_ampl = acc_integration_mem_alloc(handle->num_points_to_analyze * sizeof(*handle->lp_filt_ampl));
247  handle->unwrapped_angle = acc_integration_mem_alloc(handle->num_points_to_analyze * sizeof(*handle->unwrapped_angle));
248  handle->angle_buffer = acc_integration_mem_alloc(B_ANGLE_LENGTH * handle->num_points_to_analyze * sizeof(*handle->angle_buffer));
249  handle->filt_angle_buffer = acc_integration_mem_alloc(
250  A_ANGLE_LENGTH * handle->num_points_to_analyze * sizeof(*handle->filt_angle_buffer));
251  handle->breathing_motion_buffer =
252  acc_integration_mem_alloc(
253  handle->time_series_length * handle->num_points_to_analyze * sizeof(*handle->breathing_motion_buffer));
254  handle->hamming_window = acc_integration_mem_alloc(handle->time_series_length * sizeof(*handle->hamming_window));
255  handle->windowed_breathing_motion_buffer =
256  acc_integration_mem_alloc(handle->time_series_length * handle->num_points_to_analyze *
257  sizeof(*handle->windowed_breathing_motion_buffer));
258  handle->rfft_output =
259  acc_integration_mem_alloc(handle->rfft_output_length * handle->num_points_to_analyze * sizeof(*handle->rfft_output));
260  handle->weighted_psd = acc_integration_mem_alloc(handle->rfft_output_length * sizeof(*handle->weighted_psd));
261 
262  bool status = handle->mean_sweep != NULL && handle->filt_sparse_iq != NULL &&
263  handle->sparse_iq_buffer != NULL && handle->filt_sparse_iq_buffer != NULL && handle->angle != NULL &&
264  handle->prev_angle != NULL && handle->lp_filt_ampl != NULL && handle->unwrapped_angle != NULL &&
265  handle->angle_buffer != NULL &&
266  handle->filt_angle_buffer != NULL && handle->breathing_motion_buffer != NULL && handle->hamming_window != NULL &&
267  handle->windowed_breathing_motion_buffer != NULL && handle->rfft_output != NULL && handle->weighted_psd != NULL;
268 
269  if (status)
270  {
271  handle->freq_delta = acc_algorithm_fftfreq_delta(handle->padded_time_series_length, 1.0f / handle->frame_rate);
272  acc_algorithm_hamming(handle->time_series_length, handle->hamming_window);
273  }
274  else
275  {
276  ref_app_breathing_destroy(handle);
277  return NULL;
278  }
279  }
280 
281  return handle;
282 }

388 bool ref_app_breathing_process(ref_app_breathing_handle_t *handle, void *buffer, ref_app_breathing_result_t *result)
389 {
390  bool status = acc_detector_presence_process(handle->presence_handle, buffer, &result->presence_result);
391 
392  if (status)
393  {
394  if (result->presence_result.processing_result.calibration_needed)
395  {
396  handle->base_presence_dist = false;
397  handle->base_presence_distance = 0.0f;
398  }
399  else
400  {
401  determine_state(handle, &result->presence_result);
402 
403  update_presence_distance(handle, result->presence_result.presence_distance);
404 
405  status = perform_action_based_on_state(handle, result->presence_result.processing_result.frame, result);
406  }
407  }
408 
409  if (status)
410  {
411  result->app_state = handle->app_state;
412 
413  handle->prev_app_state = handle->app_state;
414  }
415 
416  return status;
417 }

420 static bool validate_config(ref_app_breathing_config_t *config)
421 {
422  float frame_rate = acc_detector_presence_config_frame_rate_get(config->presence_config);
423 
424  bool status = true;
425 
426  if (frame_rate == 0.0f)
427  {
428  printf("Frame rate must be set, i.e. > 0.0\n");
429  status = false;
430  }
431 
432  if (config->lowest_breathing_rate >= config->highest_breathing_rate)
433  {
434  printf("Lowest breathing rate must be lower than highest breathing rate\n");
435  status = false;
436  }
437 
438  if (config->num_dists_to_analyze < 1U)
439  {
440  printf("Number of distances to analyze must be higher than 1");
441  status = false;
442  }
443 
444  return status;
445 }

448 static void determine_state(ref_app_breathing_handle_t *handle, acc_detector_presence_result_t *presence_result)
449 {
450  if (!presence_result->presence_detected)
451  {
452  handle->app_state = REF_APP_BREATHING_APP_STATE_NO_PRESENCE;
453  }
454  else if (handle->intra_detection_threshold < presence_result->intra_presence_score)
455  {
456  handle->app_state = REF_APP_BREATHING_APP_STATE_INTRA_PRESENCE;
457  }
458  else if (!handle->base_presence_dist && handle->use_presence_processor)
459  {
460  handle->app_state = REF_APP_BREATHING_APP_STATE_DETERMINE_DISTANCE;
461  }
462  else if (!handle->use_presence_processor || handle->distance_determination_count <= handle->distance_determination_counter)
463  {
464  handle->app_state = REF_APP_BREATHING_APP_STATE_ESTIMATE_BREATHING_RATE;
465  }
466  else
467  {
468  // Do nothing
469  }
470 }

473 static void update_presence_distance(ref_app_breathing_handle_t *handle, float presence_distance)
474 {
475  if (!handle->presence_init)
476  {
477  handle->presence_init = true;
478  handle->presence_distance = presence_distance;
479  }
480 
481  handle->presence_distance = handle->presence_distance * handle->presence_sf + presence_distance * (1.0f - handle->presence_sf);
482 
483  if (handle->base_presence_dist && handle->presence_distance_threshold < fabsf(handle->base_presence_distance - handle->presence_distance))
484  {
485  handle->base_presence_dist = false;
486  handle->base_presence_distance = 0.0f;
487  }
488 }

516 static bool perform_action_based_on_state(ref_app_breathing_handle_t *handle, acc_int16_complex_t *frame, ref_app_breathing_result_t *result)
517 {
518  bool status = true;
519 
520  result->result_ready = false;
521 
522  switch (handle->app_state)
523  {
524  case REF_APP_BREATHING_APP_STATE_INIT:
525  // Do nothing
526  break;
527  // No presence and intra presence require the same action
528  case REF_APP_BREATHING_APP_STATE_NO_PRESENCE:
529  case REF_APP_BREATHING_APP_STATE_INTRA_PRESENCE:
530  handle->base_presence_dist = false;
531  handle->base_presence_distance = 0.0f;
532  break;
533  case REF_APP_BREATHING_APP_STATE_DETERMINE_DISTANCE:
534  if (handle->app_state != handle->prev_app_state)
535  {
536  handle->distance_determination_counter = 0U;
537  }
538  else
539  {
540  handle->distance_determination_counter++;
541  handle->base_presence_dist = true;
542  handle->base_presence_distance = handle->presence_distance;
543  }
544 
545  break;
546  case REF_APP_BREATHING_APP_STATE_ESTIMATE_BREATHING_RATE:
547  if (handle->app_state != handle->prev_app_state)
548  {
549  uint16_t start_p;
550  uint16_t end_p;
551 
552  if (handle->use_presence_processor)
553  {
554  uint16_t center_idx =
555  (uint16_t)(((handle->base_presence_distance - handle->start_m) / handle->step_length_m) + 0.5f);
556  start_p = center_idx >=
557  handle->num_points_to_analyze_half_width ? (uint16_t)(center_idx -
558  handle->num_points_to_analyze_half_width) : 0U;
559  end_p = (center_idx + handle->num_points_to_analyze_half_width + 1U) <=
560  handle->num_points ? (center_idx + handle->num_points_to_analyze_half_width +
561  1U) : handle->num_points;
562  }
563  else
564  {
565  start_p = 0U;
566  end_p = handle->num_points;
567  }
568 
569  reinit_breathing(handle, start_p, end_p);
570  }
571 
572  status = process_breathing(handle, frame, result);
573  break;
574  default:
575  // Should never happen
576  break;
577  }
578 
579  return status;
580 }
