23 /** \example example_detector_distance.c
24  * @brief This is an example on how the Detector Distance API can be used
25  * @n
26  * This example executes as follows:
27  * - Retrieve HAL integration
28  * - Initialize distance detector resources:
29  * + Create distance detector configuration
30  * + Update configuration settings
31  * + Create Distance detector handle
32  * + Create buffer for detector calibration data
33  * + Create buffer for sensor data
34  * - Create and calibrate the sensor
35  * - Calibrate the detector
36  * - Measure distances with the detector (loop):
37  * + Prepare sensor with the detector
38  * + Measure and wait until a read can be done
39  * + Process measurement and print the result
40  * + Handle "sensor_calibration_needed" indication
41  * - Cleanup:
42  * + Destroy detector configuration
43  * + Destroy detector handle
44  * + Destroy sensor data buffer
45  * + Destroy detector calibration data buffer
46  */

108 int acconeer_main(int argc, char *argv[])
109 {
110  (void)argc;
111  (void)argv;
112  distance_detector_resources_t resources = { 0 };
113 
114  printf("Acconeer software version %s\n", acc_version_get());
115 
116  const acc_hal_a121_t *hal = acc_hal_rss_integration_get_implementation();
117 
118  if (!acc_rss_hal_register(hal))
119  {
120  return EXIT_FAILURE;
121  }
122 
123  resources.config = acc_detector_distance_config_create();
124  if (resources.config == NULL)
125  {
126  printf("acc_detector_distance_config_create() failed\n");
127  cleanup(&resources);
128  return EXIT_FAILURE;
129  }
130 
131  set_config(resources.config, DISTANCE_PRESET_CONFIG_BALANCED);
132 
133  if (!initialize_detector_resources(&resources))
134  {
135  printf("Initializing detector resources failed\n");
136  cleanup(&resources);
137  return EXIT_FAILURE;
138  }
139 
140  // Print the configuration
141  acc_detector_distance_config_log(resources.handle, resources.config);
142 
143  /* Turn the sensor on */
144  acc_hal_integration_sensor_supply_on(SENSOR_ID);
145  acc_hal_integration_sensor_enable(SENSOR_ID);
146 
147  resources.sensor = acc_sensor_create(SENSOR_ID);
148  if (resources.sensor == NULL)
149  {
150  printf("acc_sensor_create() failed\n");
151  cleanup(&resources);
152  return EXIT_FAILURE;
153  }
154 
155  acc_cal_result_t sensor_cal_result;
156 
157  if (!do_sensor_calibration(resources.sensor, &sensor_cal_result, resources.buffer, resources.buffer_size))
158  {
159  printf("Sensor calibration failed\n");
160  cleanup(&resources);
161  return EXIT_FAILURE;
162  }
163 
164  if (!do_full_detector_calibration(&resources, &sensor_cal_result))
165  {
166  printf("Detector calibration failed\n");
167  cleanup(&resources);
168  return EXIT_FAILURE;
169  }
170 
171  while (true)
172  {
173  acc_detector_distance_result_t result = { 0 };
174 
175  if (!do_detector_get_next(&resources, &sensor_cal_result, &result))
176  {
177  printf("Could not get next result\n");
178  cleanup(&resources);
179  return EXIT_FAILURE;
180  }
181 
182  print_distance_result(&result);
183 
184  /* If "sensor calibration needed" is indicated, the sensor and detector needs to be recalibrated.
185  *
186  * Note that the detector result (printed above) is invalid if 'sensor_calibration_needed' is true.
187  */
188  if (result.sensor_calibration_needed)
189  {
190  printf("Sensor- and detector recalibration needed ... \n");
191 
192  if (!do_sensor_calibration(resources.sensor, &sensor_cal_result, resources.buffer, resources.buffer_size))
193  {
194  printf("Sensor calibration failed\n");
195  cleanup(&resources);
196  return EXIT_FAILURE;
197  }
198 
199  /* Once the sensor is recalibrated, the detector should be recalibrated and measuring can continue. */
200  if (!do_detector_recalibration(&resources, &sensor_cal_result))
201  {
202  printf("Detector recalibration failed\n");
203  cleanup(&resources);
204  return EXIT_FAILURE;
205  }
206 
207  printf("Sensor- and detector recalibration done!\n");
208  }
209  }
210 
211  cleanup(&resources);
212 
213  printf("Done!\n");
214 
215  return EXIT_SUCCESS;
216 }

277 static bool initialize_detector_resources(distance_detector_resources_t *resources)
278 {
279  resources->handle = acc_detector_distance_create(resources->config);
280  if (resources->handle == NULL)
281  {
282  printf("acc_detector_distance_create() failed\n");
283  return false;
284  }
285 
286  if (!acc_detector_distance_get_buffer_sizes(resources->handle, &(resources->buffer_size), &(resources->calibration_buffer_size)))
287  {
288  printf("acc_detector_distance_get_buffer_sizes() failed\n");
289  return false;
290  }
291 
292  resources->buffer = acc_integration_mem_alloc(resources->buffer_size);
293  if (resources->buffer == NULL)
294  {
295  printf("sensor buffer allocation failed\n");
296  return false;
297  }
298

300  if (resources->calibration_buffer == NULL)
301  {
302  printf("calibration buffer allocation failed\n");
303  return false;
304  }
305 
306  return true;
307 }

310 static bool do_sensor_calibration(acc_sensor_t *sensor,
311  acc_cal_result_t *sensor_cal_result,
312  void *buffer,
313  uint32_t buffer_size)
314 {
315  bool status = false;
316  bool cal_complete = false;
317  const uint16_t calibration_retries = 1U;
318 
319  // Random disturbances may cause the calibration to fail. At failure, retry at least once.
320  for (uint16_t i = 0; !status && (i <= calibration_retries); i++)
321  {
322  // Reset sensor before calibration by disabling/enabling it
323  acc_hal_integration_sensor_disable(SENSOR_ID);
324  acc_hal_integration_sensor_enable(SENSOR_ID);
325 
326  do
327  {
328  status = acc_sensor_calibrate(sensor, &cal_complete, sensor_cal_result, buffer, buffer_size);
329 
330  if (status && !cal_complete)
331  {
332  status = acc_hal_integration_wait_for_sensor_interrupt(SENSOR_ID, SENSOR_TIMEOUT_MS);
333  }
334  } while (status && !cal_complete);
335  }
336 
337  if (status)
338  {
339  /* Reset sensor after calibration by disabling/enabling it */
340  acc_hal_integration_sensor_disable(SENSOR_ID);
341  acc_hal_integration_sensor_enable(SENSOR_ID);
342  }
343 
344  return status;
345 }

402 static bool do_detector_get_next(distance_detector_resources_t *resources,
403  const acc_cal_result_t *sensor_cal_result,
404  acc_detector_distance_result_t *result)
405 {
406  bool result_available = false;
407 
408  do
409  {
410  if (!acc_detector_distance_prepare(resources->handle, resources->config, resources->sensor, sensor_cal_result, resources->buffer,
411  resources->buffer_size))
412  {
413  printf("acc_detector_distance_prepare() failed\n");
414  return false;
415  }
416 
417  if (!acc_sensor_measure(resources->sensor))
418  {
419  printf("acc_sensor_measure() failed\n");
420  return false;
421  }
422 
423  if (!acc_hal_integration_wait_for_sensor_interrupt(SENSOR_ID, SENSOR_TIMEOUT_MS))
424  {
425  printf("Sensor interrupt timeout\n");
426  return false;
427  }
428 
429  if (!acc_sensor_read(resources->sensor, resources->buffer, resources->buffer_size))
430  {
431  printf("acc_sensor_read() failed\n");
432  return false;
433  }
434 
435  if (!acc_detector_distance_process(resources->handle, resources->buffer, resources->calibration_buffer, &result_available,
436  result))
437  {
438  printf("acc_detector_distance_process() failed\n");
439  return false;
440  }
441  } while (!result_available);
442 
443  return true;
444 }
