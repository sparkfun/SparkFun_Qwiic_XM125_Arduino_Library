9  * @brief example_processing_ampliude.c

40 int acconeer_main(int argc, char *argv[])
41 {
42  (void)argc;
43  (void)argv;
44 
45  acc_control_helper_t control_helper_state = {0};
46 
47  printf("Acconeer software version %s\n", acc_version_get());
48 
49  const acc_hal_a121_t *hal = acc_hal_rss_integration_get_implementation();
50 
51  if (!acc_rss_hal_register(hal))
52  {
53  return EXIT_FAILURE;
54  }
55 
56  bool res = acc_control_helper_create(&control_helper_state, SENSOR_ID);
57 
58  if (!res)
59  {
60  printf("acc_control_helper_create() failed\n");
61  return EXIT_FAILURE;
62  }
63 
64  update_configuration(control_helper_state.config);
65 
66  res = acc_control_helper_activate(&control_helper_state);
67 
68  if (!res)
69  {
70  printf("acc_control_helper_activate() failed\n");
71  acc_control_helper_destroy(&control_helper_state);
72  return EXIT_FAILURE;
73  }
74 
75  uint32_t sweep_data_length = control_helper_state.proc_meta.sweep_data_length;
76 
77  acc_vector_iq_t *current_sweep = acc_vector_iq_alloc(sweep_data_length);
78  acc_vector_float_t *current_sweep_amplitude = acc_vector_float_alloc(sweep_data_length);
79 
80  bool mem_ok = (current_sweep != NULL) && (current_sweep_amplitude != NULL);
81  if (!mem_ok)
82  {
83  printf("Memory allocation for vectors failed\n");
84  }
85  else
86  {
87  uint32_t iterations = 25U;
88  for (uint32_t i = 0U; i < iterations; i++)
89  {
90  if (!acc_control_helper_get_next(&control_helper_state))
91  {
92  printf("acc_control_helper_get_next() failed\n");
93  break;
94  }
95 
96  acc_get_iq_sweep_vector(&control_helper_state, current_sweep);
97 
98  // The acc_vector_iq_amplitude function calculates the amplitude by taking
99  // absolute value of the complex valued elements in the IQ data vector.
100  // amplitude(z) = abs(z) = sqrt(real(z)^2 + imag(z)^2)
101  acc_vector_iq_amplitude(current_sweep, current_sweep_amplitude);
102 
103  acc_vector_float_print("Amplitude", current_sweep_amplitude);
104 
105  uint32_t max_peak_index = acc_vector_float_argmax(current_sweep_amplitude);
106  printf("Highest peak index: %" PRIu32 ", peak ampliude %" PRIfloat "\n",
107  max_peak_index,
108  ACC_LOG_FLOAT_TO_INTEGER(current_sweep_amplitude->data[max_peak_index]));
109  }
110  }
111 
112  acc_vector_iq_free(current_sweep);
113  acc_vector_float_free(current_sweep_amplitude);
114  acc_control_helper_destroy(&control_helper_state);
115 
116  printf("Application finished OK\n");
117 
118  return EXIT_SUCCESS;
119 }

122 static void update_configuration(acc_config_t *config)
123 {
124  int32_t start_point = 100; // start at 250 mm
125  uint16_t step_length = 2; // 2*2.5 mm = 5 mm
126  uint16_t num_points = 100; // range length 2*100*2.5 mm = 500 mm
127 
128  acc_config_start_point_set(config, start_point);
129  acc_config_num_points_set(config, num_points);
130  acc_config_step_length_set(config, step_length);
131  acc_config_profile_set(config, ACC_CONFIG_PROFILE_2);
132  acc_config_hwaas_set(config, 30);
133  // The processing in this example assumes that sweeps_per_frame = 1
134  acc_config_sweeps_per_frame_set(config, 1);
135  acc_config_prf_set(config, ACC_CONFIG_PRF_13_0_MHZ);
136 }
