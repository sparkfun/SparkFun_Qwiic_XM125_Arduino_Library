207 bool i2c_presence_detector_command(uint32_t command)
208 {
209  bool status = false;
210 
211  /* Make sure we do not have a race for i2c_detector_command/i2c_detector_status */
212  acc_integration_critical_section_enter();
213 
214  if (i2c_detector_command == 0U)
215  {
216  /* Set Ready PIN to LOW while processing the command */
217  i2c_application_system_set_ready_pin(false);
218 
219  /* Set status BUSY bit */
220  i2c_detector_status |= PRESENCE_REG_DETECTOR_STATUS_FIELD_BUSY_MASK;
221  i2c_detector_command = command;
222  status = true;
223  }
224 
225  acc_integration_critical_section_exit();
226  return status;
227 }

243 uint32_t i2c_presence_detector_get_result(void)
244 {
245  uint32_t value = 0;
246 
247  /* Make sure we do not have a race for results */
248  acc_integration_critical_section_enter();
249 
250  if (presence_detected)
251  {
252  value |= PRESENCE_REG_PRESENCE_RESULT_FIELD_PRESENCE_DETECTED_MASK;
253  }
254 
255  if (presence_detected_sticky)
256  {
257  value |= PRESENCE_REG_PRESENCE_RESULT_FIELD_PRESENCE_DETECTED_STICKY_MASK;
258 
259  presence_detected_sticky = false;
260  }
261 
262  if (presence_detector_error)
263  {
264  value |= PRESENCE_REG_PRESENCE_RESULT_FIELD_DETECTOR_ERROR_MASK;
265  }
266 
267  /* Add temperature */
268  uint32_t temp = (uint32_t)detector_resources.result.processing_result.temperature;
269 
270  temp = (temp << PRESENCE_REG_PRESENCE_RESULT_FIELD_TEMPERATURE_POS) & PRESENCE_REG_PRESENCE_RESULT_FIELD_TEMPERATURE_MASK;
271  value |= temp;
272 
273  acc_integration_critical_section_exit();
274 
275  return value;
276 }

363 int acconeer_main(int argc, char *argv[])
364 {
365  (void)argc;
366  (void)argv;
367 
368  bool setup_status = true;
369  bool get_next_frame = false;
370 
371  printf("I2C Presence Detector\n");
372  printf("Acconeer software version %s\n", acc_version_get());
373 
374  const acc_hal_a121_t *hal = acc_hal_rss_integration_get_implementation();
375 
376  if (acc_rss_hal_register(hal))
377  {
378  detector_status_set_bits(PRESENCE_REG_DETECTOR_STATUS_FIELD_RSS_REGISTER_OK_MASK);
379  }
380  else
381  {
382  printf("ERROR: acc_rss_hal_register() failed\n\n");
383  detector_status_set_bits(
384  PRESENCE_REG_DETECTOR_STATUS_FIELD_DETECTOR_ERROR_MASK | PRESENCE_REG_DETECTOR_STATUS_FIELD_RSS_REGISTER_ERROR_MASK);
385  setup_status = false;
386  }
387 
388  if (setup_status)
389  {
390  detector_resources.config = acc_detector_presence_config_create();
391  if (detector_resources.config != NULL)
392  {
393  /* Config is created, write default values to registers */
394  presence_reg_protocol_write_default();
395 
396  detector_status_set_bits(PRESENCE_REG_DETECTOR_STATUS_FIELD_CONFIG_CREATE_OK_MASK);
397  }
398  else
399  {
400  detector_status_set_bits(
401  PRESENCE_REG_DETECTOR_STATUS_FIELD_DETECTOR_ERROR_MASK | PRESENCE_REG_DETECTOR_STATUS_FIELD_CONFIG_CREATE_ERROR_MASK);
402  printf("ERROR: acc_detector_presence_config_create() failed\n\n");
403  setup_status = false;
404  }
405  }
406 
407  /* Turn the sensor on */
408  acc_hal_integration_sensor_supply_on(SENSOR_ID);
409 
410  if (setup_status)
411  {
412  /* Create sensor */
413  create_sensor(&detector_resources);
414  }
415 
416  i2c_application_system_init();
417 
418  /* Setup i2c register protocol */
419  presence_reg_protocol_setup();
420 
421  while (true)
422  {
423  /* Handle Presence Detector */
424  if (presence_detector_active &&
425  get_next_frame &&
426  !detector_status_test_bits(PRESENCE_REG_DETECTOR_STATUS_FIELD_DETECTOR_ERROR_MASK))
427  {
428  get_next_frame = false;
429  if (detector_get_next(&detector_resources))
430  {
431  print_presence_result(&detector_resources.result);
432  }
433  else
434  {
435  printf("ERROR: Could not get next result\n");
436  detector_activate(&detector_resources, false);
437  presence_detector_error = true;
438  detector_status_set_bits(PRESENCE_REG_DETECTOR_STATUS_FIELD_DETECTOR_ERROR_MASK);
439  }
440  }
441 
442  /* Handle Generic GPIO output */
443  detector_gpio_output();
444 
445  /* Handle Commands */
446  uint32_t command = get_command();
447 
448  if (command == 0)
449  {
450  /* Try to set module in low power mode */
451  module_low_power();
452 
453  /* Test if a periodic wakeup was the wakeup reason */
454  get_next_frame = i2c_application_is_periodic_wakeup();
455 
456  continue;
457  }
458 
459  /* Special command, always handle reset module command, even if error has occured */
460  if (command == PRESENCE_REG_COMMAND_ENUM_RESET_MODULE)
461  {
462  /* Reset system */
463  i2c_application_system_reset();
464  continue;
465  }
466 
467  if (detector_status_test_bits(PRESENCE_REG_DETECTOR_STATUS_FIELD_DETECTOR_ERROR_MASK))
468  {
469  /* Do not process commands after error state */
470  continue;
471  }
472 
473  /* Handle command */
474  command_handler(command);
475 
476  /* Command handler done, clear busy bit */
477  detector_status_clr_bits(PRESENCE_REG_DETECTOR_STATUS_FIELD_BUSY_MASK);
478 
479  /* Set Ready PIN to HIGH when command processing is done */
480  i2c_application_system_set_ready_pin(true);
481  }
482 
483  return EXIT_FAILURE;
484 }

492 static uint32_t get_command(void)
493 {
494  /* Make sure we do not have a race for i2c_detector_command */
495  acc_integration_critical_section_enter();
496 
497  uint32_t command = i2c_detector_command;

506 
507 static void command_handler(uint32_t command)
508 {
509  switch (command)
510  {
511  case PRESENCE_REG_COMMAND_ENUM_APPLY_CONFIGURATION:
512  if (!detector_status_test_bits(PRESENCE_REG_DETECTOR_STATUS_FIELD_CONFIG_APPLY_OK_MASK))
513  {
514  // Apply configuration
515  apply_detector_config(&detector_resources);
516  }
517 
518  break;
519  case PRESENCE_REG_COMMAND_ENUM_START_DETECTOR:
520  if (!presence_detector_active)
521  {
522  if (detector_activate(&detector_resources, true))
523  {
524  presence_detector_active = true;
525  }
526  else
527  {
528  printf("ERROR: Could not start detector\n");
529  presence_detector_error = true;
530  detector_status_set_bits(PRESENCE_REG_DETECTOR_STATUS_FIELD_DETECTOR_ERROR_MASK);
531  }
532  }
533 
534  break;
535  case PRESENCE_REG_COMMAND_ENUM_STOP_DETECTOR:
536  if (presence_detector_active)
537  {
538  if (detector_activate(&detector_resources, false))
539  {
540  presence_detector_active = false;
541  }
542  else
543  {
544  printf("ERROR: Could not stop detector\n");
545  presence_detector_error = true;
546  detector_status_set_bits(PRESENCE_REG_DETECTOR_STATUS_FIELD_DETECTOR_ERROR_MASK);
547  }
548  }
549 
550  break;
551  case PRESENCE_REG_COMMAND_ENUM_ENABLE_UART_LOGS:
552  uart_logs_enabed = true;
553  uart_log("UART logs enabled\n");
554  break;
555  case PRESENCE_REG_COMMAND_ENUM_DISABLE_UART_LOGS:
556  uart_log("UART logs disabled\n");
557  uart_logs_enabed = false;
558  break;
559  case PRESENCE_REG_COMMAND_ENUM_LOG_CONFIGURATION:
560  // Print the configuration
561  acc_detector_presence_config_log(detector_resources.config);
562  break;
563  default:
564  printf("ERROR: Unknown command: %" PRIu32 "", command);
565  break;
566  }
567 }

635 {
636  acc_hal_integration_sensor_enable(SENSOR_ID);
637 
638  detector_status_clr_bits(
639  PRESENCE_REG_DETECTOR_STATUS_FIELD_SENSOR_CALIBRATE_OK_MASK |
640  PRESENCE_REG_DETECTOR_STATUS_FIELD_SENSOR_CALIBRATE_ERROR_MASK
641  );
642

645 
646  do
647  {

649  &cal_complete,

662  printf("ERROR: acc_sensor_calibrate() failed\n");
663  }
664 
665  /* Reset sensor after calibration by disabling it */
666  acc_hal_integration_sensor_disable(SENSOR_ID);
667 
668  return status;

672 static void apply_detector_config(presence_detector_resources_t *resources)
673 {
674  bool status = true;
675 
676  acc_detector_presence_config_frame_rate_app_driven_set(resources->config, true);
677 
678  /* Always use DEEP_SLEEP for inter_frame_idle_state to save power */
679  acc_detector_presence_config_inter_frame_idle_state_set(resources->config, ACC_CONFIG_IDLE_STATE_DEEP_SLEEP);
680 
681  resources->handle = acc_detector_presence_create(resources->config, &(resources->metadata));
682  if (resources->handle != NULL)
683  {
684  detector_status_set_bits(PRESENCE_REG_DETECTOR_STATUS_FIELD_DETECTOR_CREATE_OK_MASK);
685  }
686  else
687  {
688  detector_status_set_bits(
689  PRESENCE_REG_DETECTOR_STATUS_FIELD_DETECTOR_ERROR_MASK |
690  PRESENCE_REG_DETECTOR_STATUS_FIELD_DETECTOR_CREATE_ERROR_MASK
691  );
692  printf("ERROR: acc_detector_presence_create() failed\n");
693  status = false;
694  }
695 
696  if (status)
697  {
698  if (acc_detector_presence_get_buffer_size(resources->handle, &(resources->buffer_size)))
699  {
700  detector_status_set_bits(PRESENCE_REG_DETECTOR_STATUS_FIELD_DETECTOR_BUFFER_OK_MASK);
701  }
702  else
703  {
704  detector_status_set_bits(
705  PRESENCE_REG_DETECTOR_STATUS_FIELD_DETECTOR_ERROR_MASK |
706  PRESENCE_REG_DETECTOR_STATUS_FIELD_DETECTOR_BUFFER_ERROR_MASK
707  );
708  printf("ERROR: acc_detector_presence_get_buffer_size() failed\n");
709  status = false;
710  }
711  }
712 
713  if (status)
714  {
715  resources->buffer = acc_integration_mem_alloc(resources->buffer_size);
716  if (resources->buffer != NULL)
717  {
718  detector_status_set_bits(PRESENCE_REG_DETECTOR_STATUS_FIELD_SENSOR_BUFFER_OK_MASK);
719  }
720  else
721  {
722  detector_status_set_bits(
723  PRESENCE_REG_DETECTOR_STATUS_FIELD_DETECTOR_ERROR_MASK |
724  PRESENCE_REG_DETECTOR_STATUS_FIELD_SENSOR_BUFFER_ERROR_MASK
725  );
726  printf("ERROR: sensor buffer allocation failed\n");
727  status = false;
728  }
729  }
730 
731  if (status)
732  {
733  status = calibrate_sensor(&detector_resources);
734  }
735 
736  if (status)
737  {
738  detector_status_set_bits(PRESENCE_REG_DETECTOR_STATUS_FIELD_CONFIG_APPLY_OK_MASK);
739  }
740  else
741  {
742  printf("ERROR: apply detector config failed\n");
743  detector_status_set_bits(
744  PRESENCE_REG_DETECTOR_STATUS_FIELD_DETECTOR_ERROR_MASK |
745  PRESENCE_REG_DETECTOR_STATUS_FIELD_CONFIG_APPLY_ERROR_MASK
746  );
747  }
748 }

753  bool status = true;
754 
755  if (enable)
