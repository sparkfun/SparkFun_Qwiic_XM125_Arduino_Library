180 bool i2c_distance_detector_command(uint32_t command)
181 {
182  bool status = false;
183 
184  /* Make sure we do not have a race for i2c_detector_command/i2c_detector_status */
185  acc_integration_critical_section_enter();
186 
187  if (i2c_detector_command == 0U)
188  {
189  /* Set Ready PIN to LOW while processing the command */
190  i2c_application_system_set_ready_pin(false);
191 
192  /* Set status BUSY bit */
193  i2c_detector_status |= DISTANCE_REG_DETECTOR_STATUS_FIELD_BUSY_MASK;
194  i2c_detector_command = command;
195  status = true;
196  }
197 
198  acc_integration_critical_section_exit();
199  return status;
200 }

216 uint32_t i2c_distance_detector_get_result(void)
217 {
218  uint32_t value = 0;
219 
220  /* Add number of distances */
221  value = (detector_resources.result.num_distances << DISTANCE_REG_DISTANCE_RESULT_FIELD_NUM_DISTANCES_POS) &
222  DISTANCE_REG_DISTANCE_RESULT_FIELD_NUM_DISTANCES_MASK;
223 
224  /* Add near start edge boolean */
225  if (detector_resources.result.near_start_edge_status)
226  {
227  value |= DISTANCE_REG_DISTANCE_RESULT_FIELD_NEAR_START_EDGE_MASK;
228  }
229 
230  /* Add calibration needed boolean */
231  if (detector_resources.result.sensor_calibration_needed)
232  {
233  value |= DISTANCE_REG_DISTANCE_RESULT_FIELD_CALIBRATION_NEEDED_MASK;
234  }
235 
236  if (measure_distance_error)
237  {
238  value |= DISTANCE_REG_DISTANCE_RESULT_FIELD_MEASURE_DISTANCE_ERROR_MASK;
239  }
240 
241  /* Add temperature */
242  uint32_t temp = (uint32_t)detector_resources.result.temperature;
243 
244  temp = (temp << DISTANCE_REG_DISTANCE_RESULT_FIELD_TEMPERATURE_POS) & DISTANCE_REG_DISTANCE_RESULT_FIELD_TEMPERATURE_MASK;
245  value |= temp;
246 
247  return value;
248 }

322 int acconeer_main(int argc, char *argv[])
323 {
324  (void)argc;
325  (void)argv;
326 
327  bool setup_status = true;
328 
329  printf("I2C Distance Detector\n");
330  printf("Acconeer software version %s\n", acc_version_get());
331 
332  const acc_hal_a121_t *hal = acc_hal_rss_integration_get_implementation();
333 
334  if (acc_rss_hal_register(hal))
335  {
336  detector_status_set_bits(DISTANCE_REG_DETECTOR_STATUS_FIELD_RSS_REGISTER_OK_MASK);
337  }
338  else
339  {
340  printf("ERROR: acc_rss_hal_register() failed\n\n");
341  detector_status_set_bits(
342  DISTANCE_REG_DETECTOR_STATUS_FIELD_DETECTOR_ERROR_MASK | DISTANCE_REG_DETECTOR_STATUS_FIELD_RSS_REGISTER_ERROR_MASK);
343  setup_status = false;
344  }
345 
346  if (setup_status)
347  {
348  detector_resources.config = acc_detector_distance_config_create();
349  if (detector_resources.config != NULL)
350  {
351  /* Config is created, write default values to registers */
352  distance_reg_protocol_write_default();
353 
354  detector_status_set_bits(DISTANCE_REG_DETECTOR_STATUS_FIELD_CONFIG_CREATE_OK_MASK);
355  }
356  else
357  {
358  detector_status_set_bits(
359  DISTANCE_REG_DETECTOR_STATUS_FIELD_DETECTOR_ERROR_MASK | DISTANCE_REG_DETECTOR_STATUS_FIELD_CONFIG_CREATE_ERROR_MASK);
360  printf("ERROR: acc_detector_distance_config_create() failed\n\n");
361  setup_status = false;
362  }
363  }
364 
365  /* Turn the sensor on */
366  acc_hal_integration_sensor_supply_on(SENSOR_ID);
367 
368  if (setup_status)
369  {
370  /* Create sensor */
371  create_sensor(&detector_resources);
372  }
373 
374  i2c_application_system_init();
375 
376  /* Setup i2c register protocol */
377  distance_reg_protocol_setup();
378 
379  while (true)
380  {
381  uint32_t command = get_command();
382 
383  if (command == 0)
384  {
385  if (i2c_application_system_test_wakeup_pin())
386  {
387  i2c_application_system_wait_for_interrupt();
388  }
389  else
390  {
391  /* Set ready pin LOW, we are about to power down */
392  i2c_application_system_set_ready_pin(false);
393 
394  uart_log("Enter low power state\n");
395  i2c_application_enter_low_power_state();
396  uart_log("Exit low power state\n");
397 
398  /**
399  * Test if measure on wake up is enable
400  * Do a measurement if the the detector is ready (configured and calibrated)
401  */
402  if (is_detector_ready() && i2c_distance_detector_get_measure_on_wakeup())
403  {
404  uart_log("Measure on wakeup\n");
405  command = DISTANCE_REG_COMMAND_ENUM_MEASURE_DISTANCE;
406  }
407  }
408 
409  /* Do an extra check on command, are we in measure on wakeup mode? */
410  if (command == 0)
411  {
412  /* Set ready pin HIGH, we are ready for communication */
413  i2c_application_system_set_ready_pin(true);
414 
415  /* No command to process */
416  continue;
417  }
418  }
419 
420  /* Special command, always handle reset module command, even if error has occured */
421  if (command == DISTANCE_REG_COMMAND_ENUM_RESET_MODULE)
422  {
423  /* Reset system */
424  i2c_application_system_reset();
425  continue;
426  }
427 
428  if (detector_status_test_bits(DISTANCE_REG_DETECTOR_STATUS_FIELD_DETECTOR_ERROR_MASK))
429  {
430  /* Do not process commands after error state */
431  continue;
432  }
433 
434  /* Handle command */
435  command_handler(command);
436 
437  /* Command handler done, clear busy bit */
438  detector_status_clr_bits(DISTANCE_REG_DETECTOR_STATUS_FIELD_BUSY_MASK);
439 
440  /* Set Ready PIN to HIGH when command processing is done */
441  i2c_application_system_set_ready_pin(true);
442  }
443 
444  return EXIT_FAILURE;
445 }

467 
468 static void command_handler(uint32_t command)
469 {
470  bool do_apply_config = false;
471  bool do_calibrate = false;
472  bool do_recalibrate = false;
473 
474  switch (command)
475  {
476  case DISTANCE_REG_COMMAND_ENUM_APPLY_CONFIGURATION:
477  if (!detector_status_test_bits(DISTANCE_REG_DETECTOR_STATUS_FIELD_CONFIG_APPLY_OK_MASK))
478  {
479  do_apply_config = true;
480  }
481 
482  break;
483  case DISTANCE_REG_COMMAND_ENUM_CALIBRATE:
484  if (detector_status_test_bits(DISTANCE_REG_DETECTOR_STATUS_FIELD_CONFIG_APPLY_OK_MASK))
485  {
486  do_calibrate = true;
487  }
488 
489  break;
490  case DISTANCE_REG_COMMAND_ENUM_RECALIBRATE:
491  if (detector_status_test_bits(DISTANCE_REG_DETECTOR_STATUS_FIELD_CONFIG_APPLY_OK_MASK))
492  {
493  do_recalibrate = true;
494  }
495 
496  break;
497  case DISTANCE_REG_COMMAND_ENUM_APPLY_CONFIG_AND_CALIBRATE:
498  if (!detector_status_test_bits(DISTANCE_REG_DETECTOR_STATUS_FIELD_CONFIG_APPLY_OK_MASK))
499  {
500  do_apply_config = true;
501  do_calibrate = true;
502  }
503 
504  break;
505  case DISTANCE_REG_COMMAND_ENUM_MEASURE_DISTANCE:
506  memset(&detector_resources.result, 0, sizeof(detector_resources.result));
507 
508  /* Increase counter - Make sure we do not have a race for measure_distance_counter */
509  acc_integration_critical_section_enter();
510  measure_distance_counter++;
511  acc_integration_critical_section_exit();
512 
513  if (detector_get_next(&detector_resources))
514  {
515  print_distance_result(&detector_resources.result);
516  measure_distance_error = false;
517  }
518  else
519  {
520  printf("ERROR: Could not get next result\n");
521  measure_distance_error = true;
522  }
523 
524  break;
525  case DISTANCE_REG_COMMAND_ENUM_ENABLE_UART_LOGS:
526  uart_logs_enabed = true;
527  uart_log("UART logs enabled\n");
528  break;
529  case DISTANCE_REG_COMMAND_ENUM_DISABLE_UART_LOGS:
530  uart_log("UART logs disabled\n");
531  uart_logs_enabed = false;
532  break;
533  case DISTANCE_REG_COMMAND_ENUM_LOG_CONFIGURATION:
534  // Print the configuration
535  acc_detector_distance_config_log(detector_resources.handle, detector_resources.config);
536  break;
537  default:
538  printf("ERROR: Unknown command: %" PRIu32 "", command);
539  break;
540  }
541 
542  /* Apply config is activated by APPLY_CONFIGURATION or APPLY_CONFIG_AND_CALIBRATE */
543  if (do_apply_config)
544  {
545  apply_detector_config(&detector_resources);
546  }
547 
548  if (do_calibrate || do_recalibrate)
549  {
550  calibrate_sensor(&detector_resources);
551  calibrate_detector(&detector_resources, do_recalibrate);
552  }
553 }

620 static void apply_detector_config(distance_detector_resources_t *resources)
621 {
622  bool status = true;
623 
624  resources->handle = acc_detector_distance_create(resources->config);
625  if (resources->handle != NULL)
626  {
627  detector_status_set_bits(DISTANCE_REG_DETECTOR_STATUS_FIELD_DETECTOR_CREATE_OK_MASK);
628  }
629  else
630  {
631  detector_status_set_bits(
632  DISTANCE_REG_DETECTOR_STATUS_FIELD_DETECTOR_ERROR_MASK |
633  DISTANCE_REG_DETECTOR_STATUS_FIELD_DETECTOR_CREATE_ERROR_MASK
634  );
635  printf("ERROR: acc_detector_distance_create() failed\n");
636  status = false;
637  }
638 
639  if (status)
640  {
641  if (acc_detector_distance_get_buffer_sizes(resources->handle, &(resources->buffer_size), &(resources->calibration_buffer_size)))
642  {
643  detector_status_set_bits(DISTANCE_REG_DETECTOR_STATUS_FIELD_DETECTOR_BUFFER_OK_MASK);
644  }
645  else
646  {
647  detector_status_set_bits(
648  DISTANCE_REG_DETECTOR_STATUS_FIELD_DETECTOR_ERROR_MASK |
649  DISTANCE_REG_DETECTOR_STATUS_FIELD_DETECTOR_BUFFER_ERROR_MASK
650  );
651  printf("ERROR: acc_detector_distance_get_buffer_sizes() failed\n");
652  status = false;
653  }
654  }
655 
656  if (status)
657  {
658  resources->buffer = acc_integration_mem_alloc(resources->buffer_size);
659  if (resources->buffer != NULL)
660  {
661  detector_status_set_bits(DISTANCE_REG_DETECTOR_STATUS_FIELD_SENSOR_BUFFER_OK_MASK);
662  }
663  else
664  {
665  detector_status_set_bits(
666  DISTANCE_REG_DETECTOR_STATUS_FIELD_DETECTOR_ERROR_MASK |
667  DISTANCE_REG_DETECTOR_STATUS_FIELD_SENSOR_BUFFER_ERROR_MASK
668  );
669  printf("ERROR: sensor buffer allocation failed\n");
670  status = false;
671  }
672  }
673 
674  if (status)
675  {
676  resources->calibration_buffer = acc_integration_mem_alloc(resources->calibration_buffer_size);
677  if (resources->calibration_buffer != NULL)
678  {
679  detector_status_set_bits(DISTANCE_REG_DETECTOR_STATUS_FIELD_CALIBRATION_BUFFER_OK_MASK);
680  }
681  else
682  {
683  detector_status_set_bits(
684  DISTANCE_REG_DETECTOR_STATUS_FIELD_DETECTOR_ERROR_MASK |
685  DISTANCE_REG_DETECTOR_STATUS_FIELD_CALIBRATION_BUFFER_ERROR_MASK
686  );
687  printf("ERROR: calibration buffer allocation failed\n");
688  status = false;
689  }
690  }
691 
692  if (status)
693  {
694  detector_status_set_bits(DISTANCE_REG_DETECTOR_STATUS_FIELD_CONFIG_APPLY_OK_MASK);
695  }
696  else
697  {
698  printf("ERROR: apply detector config failed\n");
699  detector_status_set_bits(
700  DISTANCE_REG_DETECTOR_STATUS_FIELD_DETECTOR_ERROR_MASK |
701  DISTANCE_REG_DETECTOR_STATUS_FIELD_CONFIG_APPLY_ERROR_MASK
702  );
703  }

707 static void calibrate_sensor(distance_detector_resources_t *resources)
708 {
709  acc_hal_integration_sensor_enable(SENSOR_ID);
710 
711  detector_status_clr_bits(
712  DISTANCE_REG_DETECTOR_STATUS_FIELD_SENSOR_CALIBRATE_OK_MASK |
713  DISTANCE_REG_DETECTOR_STATUS_FIELD_SENSOR_CALIBRATE_ERROR_MASK
714  );
715 
716  bool status;
717  bool cal_complete = false;
718 
719  do
720  {
721  status = acc_sensor_calibrate(resources->sensor,
722  &cal_complete,
723  &resources->sensor_cal_result,
724  resources->buffer,
725  resources->buffer_size);
726  } while (status && !cal_complete && acc_hal_integration_wait_for_sensor_interrupt(SENSOR_ID, SENSOR_TIMEOUT_MS));
727 
728  if (status)
729  {
730  detector_status_set_bits(DISTANCE_REG_DETECTOR_STATUS_FIELD_SENSOR_CALIBRATE_OK_MASK);
731  }
732  else
733  {
